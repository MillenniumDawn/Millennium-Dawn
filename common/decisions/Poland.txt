		fire_only_once = no
		days_re_enable = 120
		cost = 50

		complete_effect = {
			CZE = {
				IF = {
					limit = {
						NOT = {
							has_country_flag = received_pol_influence
						}
					}

					country_event = poland.influence.1
				}
				else_if = {

					nationalist_right_wing_populists_are_in_power = yes
					nationalist_fascist_are_in_power = yes
					is_in_faction_with = POL
				}
			}
		}
