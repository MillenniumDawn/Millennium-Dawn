#Shows what the rise will be and sets flags
display_opinion_rise_small_medium_business_owners = {
	if = {
		limit = { 
				has_country_flag = hostile_small_medium_business_owners 
				NOT = { has_country_flag = opinion_freze }
				}
		increase_hostile = yes #display
		clr_country_flag = hostile_small_medium_business_owners
		set_country_flag = negative_small_medium_business_owners
		set_country_flag = opinion_freze #exit
	}
	if = {
		limit = { 
				has_country_flag = negative_small_medium_business_owners 
				NOT = { has_country_flag = opinion_freze }
				}
		increase_negative = yes #display
		clr_country_flag = negative_small_medium_business_owners
		set_country_flag = indifferent_small_medium_business_owners
		set_country_flag = opinion_freze #exit
	}
	if = {
		limit = { 
				has_country_flag = indifferent_small_medium_business_owners 
				NOT = { has_country_flag = opinion_freze }
				}
		increase_indifferent = yes #display
		clr_country_flag = indifferent_small_medium_business_owners
		set_country_flag = positive_small_medium_business_owners
		set_country_flag = opinion_freze #exit
	}
	if = {
		limit = { 
				has_country_flag = positive_small_medium_business_owners 
				NOT = { has_country_flag = opinion_freze }
				}
		increase_positive = yes #display
		clr_country_flag = positive_small_medium_business_owners
		set_country_flag = enthusiastic_small_medium_business_owners
		set_country_flag = opinion_freze #exit
	}
	if = {
		limit = { 
				has_country_flag = enthusiastic_small_medium_business_owners 
				NOT = { has_country_flag = opinion_freze }
				}
		add_political_power = 100
		set_country_flag = opinion_freze #exit
	}
	clr_country_flag = opinion_freze
}
