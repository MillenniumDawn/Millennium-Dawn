	on_monthly_UOE = {
		effect = {
			#updates the states with below or above average productivity
			if = {
				limit = {
					NOT = {
						has_country_flag = resolved_eco_crisis
						has_country_flag = states_exited_due_eco

			}

				UOE_mil_revolt_track = yes
			}
		}
	}
