decay_mdmosjap_conrad_left = {
	set_temp_variable = { mdmosjap_JAP_temp_var_LeftRadDecayed = mdmosjap_JAP_var_LeftRad }
	set_variable = { mdmosjap_JAP_var_LeftRadDecay = -0.2 }	#Base Decay

	if = {
		#JAP has SASPL idea
		limit = {
			ROOT = {
				OR = {
					has_idea = JAP_saspl_weak
					has_idea = JAP_saspl_middle
					has_idea = JAP_saspl_strong
				}
				NOT = { has_idea = JAP_sealds }
			}
		}
		add_to_variable = { mdmosjap_JAP_var_LeftRadDecay = 0.1 }
	} else_if = {
		#JAP has SEALDs idea
		limit = {
			ROOT = {
				AND = {
					NOT = { has_idea = JAP_saspl_weak }
					NOT = { has_idea = JAP_saspl_middle }
					NOT = { has_idea = JAP_saspl_strong }
				}
				has_idea = JAP_sealds
			}
		}
		add_to_variable = { mdmosjap_JAP_var_LeftRadDecay = 0.2 }
	}
	if = {
		#JAP's Leader has Cyber Left Wings idea
		limit = {
			ROOT = {
				has_country_leader_with_trait = sayoku_supporter
			}
		}
		add_to_variable = { mdmosjap_JAP_var_LeftRadDecay = 0.1 }
	}
	if = {
		#JAP has 3.11 idea but after the Fukushima Disaster
		limit = {
			ROOT = {
				OR = {
					has_idea = JAP_311
					has_idea = JAP_311_tier2
				}
			}
		}
		add_to_variable = { mdmosjap_JAP_var_LeftRadDecay = 0.3 }
	}
	if = {
		#JAP has 3.11 idea but having established the Reconstruction Authority
		limit = {
			ROOT = {
				has_idea = JAP_311_tier3
			}
		}
		add_to_variable = { mdmosjap_JAP_var_LeftRadDecay = 0.2 }
	}
	if = {
		#JAP has Memory of 3.11 idea
		limit = {
			ROOT = {
				OR = {
					has_idea = JAP_aftermath_of_311
					has_idea = JAP_aftermath_of_311_ndr
					has_idea = JAP_aftermath_of_311_ndr_nra
				}
			}
		}
		add_to_variable = { mdmosjap_JAP_var_LeftRadDecay = 0.1 }
	}

	add_to_temp_variable = { mdmosjap_JAP_temp_var_LeftRadDecayed = mdmosjap_JAP_var_LeftRadDecay }
	set_variable = { mdmosjap_JAP_var_LeftRad = mdmosjap_JAP_temp_var_LeftRadDecayed }

	CALC_MDMOSJAP_CONRAD_MAXLIMIT = yes
}

	ROOT = {
		#Copied and arranged from spread_influence
		add_to_variable = { influence_gain = 10 }
		for_loop_effect = {
			end = influence_array^num
			value = v
				if = {
				limit = {
						check_variable = { influence_array^v = FROM }
				}
				add_to_variable = { influence_array_val^v = influence_gain }
				set_country_flag = found
			}
		}
		if = {
				limit = { NOT = { has_country_flag = found } }
				add_to_array = { influence_array = FROM.id }
				add_to_array = { influence_array_val = influence_gain }
		}
		clr_country_flag = found
		recalculate_influence = yes
	}
}

					OR = {

						}
						is_in_array = { ruling_party = 0 }
						is_in_array = { ruling_party = 6 }
						is_in_array = { ruling_party = 7 }
						is_in_array = { ruling_party = 15 }

						is_in_array = { ruling_party = 21 }
						is_in_array = { ruling_party = 22 }
						is_in_array = { ruling_party = 23 }
					}
					AND = {
						NOT = { has_country_leader = { name = "Yasuo Fukuda" } }
						NOT = { has_country_leader = { name = "Yoshihide Suga" } }
						NOT = { has_country_leader = { name = "Fumio Kishida" } }
						NOT = { has_country_leader = { name = "Yohei Kono" } }
						NOT = { has_country_leader = { name = "Hiromu Nonaka" } }

				}

#The Crisis of Autumn Equinox Calculation
#--------------------------------------------------------------
calc_mdmosjap_kato_rebellion_epilogue_kono_ending_crisis_result = {
	set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_L_absolute = 0 }
	set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_R_absolute = 0 }
	set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = 0 }
	if = {	#I am not good at mathematics since 7th grade :(
		limit = { check_variable = { mdmosjap_var_shubun_kiki_point_L = mdmosjap_var_shubun_kiki_point_R } }
		set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = 0 }
	} else_if = {
		limit = { check_variable = { mdmosjap_var_shubun_kiki_point_L > mdmosjap_var_shubun_kiki_point_R } }
		if = {
			limit = {
				OR = {
					check_variable = { mdmosjap_var_shubun_kiki_point_R > 0 }		#ex)  4 v 2  (2)
					check_variable = { mdmosjap_var_shubun_kiki_point_R = 0 }		#ex)  4 v 0  (4)
				}
			}
			set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_L }
			subtract_from_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_R }
		} else_if = {
			limit = {
				check_variable = { mdmosjap_var_shubun_kiki_point_R < 0 }
			}
			if = {
				limit = {
					OR = {
						check_variable = { mdmosjap_var_shubun_kiki_point_L > 0 }	#ex)  2 v -2 (4)
						check_variable = { mdmosjap_var_shubun_kiki_point_L = 0 }	#ex)  0 v -2 (2)
					}
				}
				set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_L }
				set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_R_absolute = mdmosjap_var_shubun_kiki_point_R }
				multiply_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_R_absolute = -1 }
				add_to_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_temp_var_shubun_kiki_point_R_absolute }
			} else_if = {
				limit = {
					check_variable = { mdmosjap_var_shubun_kiki_point_L < 0 }	#ex) -1 v -2 (1)
				}
				set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_L }
				subtract_from_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_R }
			}
		}
	} else_if = {
		limit = { check_variable = { mdmosjap_var_shubun_kiki_point_L < mdmosjap_var_shubun_kiki_point_R } }
		if = {
			limit = {
				OR = {
					check_variable = { mdmosjap_var_shubun_kiki_point_L > 0 }		#ex)  2 v 4  (2)
					check_variable = { mdmosjap_var_shubun_kiki_point_L = 0 }		#ex)  0 v 4  (4)
				}
			}
			set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_R }
			subtract_from_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_L }
		} else_if = {
			limit = {
				check_variable = { mdmosjap_var_shubun_kiki_point_L < 0 }
			}
			if = {
				limit = {
					OR = {
						check_variable = { mdmosjap_var_shubun_kiki_point_R > 0 }	#ex) -2 v 2  (4)
						check_variable = { mdmosjap_var_shubun_kiki_point_R = 0 }	#ex) -2 v 0  (2)
					}
				}
				set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_R }
				set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_L_absolute = mdmosjap_var_shubun_kiki_point_L }
				multiply_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_L_absolute = -1 }
				add_to_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_temp_var_shubun_kiki_point_L_absolute }
			} else_if = {
				limit = {
					check_variable = { mdmosjap_var_shubun_kiki_point_R < 0 }	#ex) -2 v -1 (1)
				}
				set_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_R }
				subtract_from_temp_variable = { mdmosjap_temp_var_shubun_kiki_point_difference_absolute = mdmosjap_var_shubun_kiki_point_L }
			}
		}
	}
	set_variable = { mdmosjap_var_shubun_kiki_point_difference = mdmosjap_temp_var_shubun_kiki_point_difference_absolute }
}
