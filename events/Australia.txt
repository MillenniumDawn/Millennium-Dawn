        name = australia_md.2.o1

        modify_treasury_effect = yes
        log = "[GetDateText]: [This.GetName]: australia_md.2.o1 executed"

			}

				limit = { has_dlc = "Man the Guns" }

			}
			else = {

			}
        }
    }
}
