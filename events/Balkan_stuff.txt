			country_event = {
				id = KFOR.101
				days = 4
			}
		}
		SER = {
			country_event = {
			id = KFOR.105
			days = 4
			}
		}
		set_party_index_to_ruling_party = yes
		set_temp_variable = { party_popularity_increase = 0.025 }
		set_temp_variable = { temp_outlook_increase = 0.015 }

			base = 2

			add_opinion_modifier = { target = SER modifier = note_of_protest }

		GER = {
			add_opinion_modifier = { target = SOV modifier = note_of_protest }
		}
		set_party_index_to_ruling_party = yes
