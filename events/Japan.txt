#Disputed Islands Mediation: Conference Gate
#Gate, chooses the next event to serve until done
country_event = {
	id = Japan.100009310
	title = Japan.100009310.t
	desc = Japan.100009310.d
	picture = GFX_JAP_diet_address

	is_triggered_only = yes

	immediate = {
		clear_variable = jap_mediation_conference_side0
		clear_variable = jap_mediation_conference_side1
		clear_variable = jap_mediation_conference_side2
		clear_variable = jap_mediation_conference_favored_candidate
		clear_array = jap_mediation_conference_transfer_states
	}

	option = {
		name = Japan.100009310.oA
		log = "[GetDateText]: [This.GetName]: Japan.100009310.oA executed"
		if = { # this event loop will proceed so long as there are unresolved disputes left (spratlys is a special case)
			limit = {
				check_variable = { jap_mediation_conference_states^num > 0 }
			}
			set_temp_variable = { conference_target_state = jap_mediation_conference_states^0 }
			#add any states for this round to the transfer states array
			add_to_array = { jap_mediation_conference_transfer_states = conference_target_state }
			#set the sides for this round of negotiation
			var:conference_target_state = {
				owner = {
					set_temp_variable = { conference_side0 = THIS }
					every_other_country = { #we only want OTHER tags that have claims, not the owner
						limit = {
							var:conference_target_state = {
								OR = {
									is_claimed_by = PREV
									is_core_of = PREV
								}
							}
							is_in_array = {
								array = JAP.jap_mediation_conference_attendees
								value = THIS
							}
						}
						add_to_temp_array = { conference_current_claimants = THIS }
					}
				}
			}
			set_variable = { jap_mediation_conference_side0 = conference_side0 }
			#select the first two non-owner claimants and discard the rest to keep the list reasonable
			if = {
				limit = {
					check_variable = { conference_current_claimants^num > 1 }
				}
				set_variable = { jap_mediation_conference_side2 = conference_current_claimants^1 }
			}
			if = {
				limit = {
					check_variable = { conference_current_claimants^num > 0 }
				}
				set_variable = { jap_mediation_conference_side1 = conference_current_claimants^0 }
				# if we have at least one opposing side attending, we can attempt to resolve
				country_event = Japan.100009311 #Dispute Resolution Draft
			}
			else = {
				remove_from_array = { #pop the first element in the conference list so the next cycle knows what to do
					array = jap_mediation_conference_states
					index = 0
				}
				country_event = Japan.100009310 #Return to gate
			}
			remove_from_array = { #pop the first element in the conference list so the next cycle knows what to do
				array = jap_mediation_conference_states
				index = 0
			}
		}
		else_if = { # spratlys not yet resolved

		}
		else = { # no more disputes - end conference

		}
	}
}
