#ethiopia requests Investments from China
country_event = {
	id = ethiopia.1
	title = ethiopia.1.t
	desc = ethiopia.1.d
	picture = GFX_united_states_dollar

	is_triggered_only = yes

	option = {
		name = ethiopia.1.a
		log = "[GetDateText]: [This.GetName]: ethiopia.1.a executed"
		add_political_power = -10
		ETH = {
			add_political_power = 30
			add_opinion_modifier = {
				target = ROOT
				modifier = loan_granted
			}
			add_to_variable = { influencer2_amount = 5 }
			recalculate_influence = yes
		}

		ai_chance = {
			factor = 200
		}
	}

	option = {
		name = ethiopia.1.b
		log = "[GetDateText]: [This.GetName]: ethiopia.1.b executed"
		add_popularity = {
			ideology = democratic
			popularity = 0.01
		}
		add_to_variable = { liberalism_pop = 0.01 }
		recalculate_party = yes

		ETH = {
			add_opinion_modifier = {
				target = ROOT
				modifier = loan_denied
			}
			add_to_variable = { influencer2_amount = -5 }
			recalculate_influence = yes
		}

		ai_chance = {
			factor = 15
		}
	}
}
