scripted_gui = {

	# The button to acutally open up the UI
	open_the_tutorial_gui = {
		context_type = player_context
		window_name = "tutorial_open_scripted_gui_window"

		visible = {
			is_ai = no
		}

		triggers = { open_the_tutorial_gui_click_enabled = { is_ai = no } }

		effects = {
			open_the_tutorial_gui_click = {
				tutorial_screen_selection = yes
				if = { limit = { NOT = { has_country_flag = new_tutorial_open } }
					set_country_flag = new_tutorial_open
				}
				else = {
					clr_country_flag = new_tutorial_open
				}
				set_country_flag = show_politics_tutorial_screen
			}
		}
	}

	tutorial_gui = {
		context_type = player_context
		window_name = "tutorial_main_background_window"

				has_country_flag = show_economy_tutorial_gdp
			}
		}

				tutorial_screen_selection = yes

			}

			}
			
			military_tutorial_screen_click = {
				tutorial_screen_selection = yes

				set_country_flag = show_military_tutorial_screen
			}

				tutorial_screen_selection = yes
				economy_tutorial_selection = yes
				set_country_flag = show_diplomacy_tutorial_screen
			}

			others_tutorial_screen_click = {
				tutorial_screen_selection = yes
				economy_tutorial_selection = yes
				set_country_flag = show_others_tutorial_screen
			}

			#Economy
			economy_tutorial_ui_click = {
				economy_tutorial_selection = yes
				set_country_flag = show_economy_tutorial_ui
			}
