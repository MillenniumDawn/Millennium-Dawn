##Picking up the topic
event = {
	id = 7771001
	random = no
	country = ARG
	
	trigger = {
		NOT = {
			control = { province = 866 data = ARG }
			control = { province = 866 data = U06 } #wont go against whole UE
			alliance = { country = ARG country = BRA } #this would spoil the fun
		}
		atwar = no #not at war
		army = 6 #at least 6 land units, starting 3 divisions
	 	OR = { 
			domestic = { type = interventionism value = 8 }	#at least somewhat aggresive
	 		random = 10		#or random
			atwar = ENG		#or use the fact that they're in war
		}
	}
	
	date = { day = 2 month = april year = 2007 } #25 years from the first conflict + random 30 days
	offset = 30	#if not at first, then repeat check every month
	deathdate = { day = 1 month = july year = 2012 } #30 years from end of conflict
	
	name = "The Falkland Question" 
	desc = "Over 25 years have passed since the Falkland Conflict. Still, we consider this to be our territory, not a British one. Maybe it is time to open the topic again?"
	picture = "shipport"
	style = 0
	
	
	action_a = {
		name = "Let's talk."
	ai_chance = 99
		command = { type = money value = -10 }
		command = { type = trigger which = 7771002 }
	}

	action_b = {
		name = "We are not really interested anyway"
	ai_chance = 1
		command = { type = removecore which = 866 }
	}
}
