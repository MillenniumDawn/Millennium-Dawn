	picture = GFX_news_event_cole_attack
	
	major = yes
	
	is_triggered_only = yes
	
	option = {
		name = usa_news.18.o1
		log = "[GetDateText]: [This.GetName]: usa_news.18.o1 executed"
		trigger = {
			original_tag = USA
		}
	}
	
	option = {
		name = usa_news.18.o2
		log = "[GetDateText]: [This.GetName]: usa_news.18.o2 executed"
		trigger = {
			NOT = { original_tag = USA }
			NOT = { original_tag = NKO }
			NOT = { original_tag = SUD }
			NOT = { has_war_with = USA }
		}
	}
	
	option = {
		name = usa_news.18.o3
		log = "[GetDateText]: [This.GetName]: usa_news.18.o3 executed"
		trigger = {
			OR = {
				original_tag = NKO
				has_war_with = USA
			}
		}
	}
	
	option = {
		name = usa_news.18.o4
		log = "[GetDateText]: [This.GetName]: usa_news.18.o4 executed"
		trigger = {
			original_tag = SUD
		}
		effect_tooltip = {
			USA = {
				add_opinion_modifier = {
					target = SUD
					modifier = recent_actions_negative
				}
			}
		}
	}
}
