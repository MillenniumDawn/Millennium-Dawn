## Alcohol Abuse Change
country_event = {
	id = greenland.1
	title = greenland.1.t
	desc = greenland.1.d
	picture = GFX_politics_protest
	is_triggered_only = yes

	trigger = {
		tag = GRN
		check_variable = {
			GRN_modify_alcohol_abuse = yes
		}
	}

	option = {
		name = greenland.1.a
		custom_effect_tooltip = GRN_modify_alcohol_abuse_tt
	}
}

## General Wellness Change
country_event = {
	id = greenland.3
	title = greenland.3.t
	desc = greenland.3.d
	picture = GFX_politics_protest
	is_triggered_only = yes

	trigger = {
		tag = GRN
		check_variable = {
			AND = {
				GRN_modify_alcohol_abuse = yes
				GRN_modify_unalive = yes
			}
		}
	}

	option = {
		name = greenland.3.a
		custom_effect_tooltip = GRN_modify_alcohol_abuse_tt
		custom_effect_tooltip = GRN_modify_unalive_tt
	}
}

country_event = {
	id = greenland_agri.1
	title = greenland_agri.1.t
	desc = greenland_agri.1.d
	picture = GFX_politics_protest
	is_triggered_only = yes

	option = {
		name = greenland_agri.1.a
		swap_ideas = {
			remove_idea = GRN_Local_Production_Advancements
			add_idea = GRN_Local_Production_Advancements_2L
		}
		set_temp_variable = { modify_alcohol_abuse = -5 }
		GRN_modify_alcohol_abuse = yes
	}

	option = {
		name = greenland_agri.1.b
		swap_ideas = {
			remove_idea = GRN_Local_Production_Advancements
			add_idea = GRN_Local_Production_Advancements_2M
		}
	}

	option = {
		name = greenland_agri.1.c
		swap_ideas = {
			remove_idea = GRN_Local_Production_Advancements
			add_idea = GRN_Local_Production_Advancements_2H
		}
		set_temp_variable = { modify_alcohol_abuse = 5 }
		GRN_modify_alcohol_abuse = yes
	}
}
