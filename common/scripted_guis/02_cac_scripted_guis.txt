scripted_gui = {
	companies_selecting_bar = {
		window_name = "companies_selecting_bar"
        context_type = player_context
		parent_window_token = politics_tab

		properties = {
			companies_slot_grid_item_btn = {
				image = "[MD_cac_icon_of_company]"
			}
		}

        dynamic_lists = {
            companies_slot_grid = {
                array = MD_cac_companies_slots
                entry_container = "companies_slot_grid_item"
			}
		}

		effects = {
            companies_slot_grid_item_btn_click = {
				set_variable = { MD_cac_selected_company = v }
				set_variable = { MD_cac_selected_index = i }

                if = {
                    limit = {
                        check_variable = { MD_cac_companies_slots^i < 0 }
                    }
                    set_variable = { opened_companies_window = i }
                    set_variable = { opened_company_window = -1 }
                }
                else = {
                    set_variable = { opened_company_window = i }
                    set_variable = { opened_companies_window = -1 }
                }
            }
		}
	}

    companies_selecting_window = {
		window_name = "companies_selecting_list_window"
        context_type = player_context
		parent_window_token = politics_tab

        visible = {
			ROOT = {
				OR = {
					is_ai = yes
					check_variable = { opened_companies_window > -1 }

			}
		}

            }
        }

		triggers = {
			avaible_companies_select_click_enabled = {
				set_temp_variable = { company_cost_temp = 1 }
				add_to_temp_variable = { company_cost_temp = modifier@companies_cost_modifier } 
				multiply_temp_variable = { company_cost_temp = GLOBAL.MD_cac_companies_cost^v }
				
				check_variable = { political_power > company_cost_temp }
			}
		}

		effects = {
			avaible_companies_select_click = {
				set_temp_variable = { cac_selected_company = v }
				MD_cac_select_company_effect = yes

				remove_from_array = {
					array = MD_cac_country_avaibles_companies
					index = i
				}

				set_variable = { opened_companies_window = -1 }
			}

			close_click = {
                set_variable = { opened_companies_window = -1 }
            }
		}
	}
