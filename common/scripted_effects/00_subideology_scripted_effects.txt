# stores the info of popularities. modified version of store_election_values = yes
hold_elections = {
	add_up_party_totals = yes
	set_party_positions = yes
	for_loop_effect = {
		end = party_pop_array^num
		value = v
		if = {
			limit = {
				#normal popularity entry is over threshold
				check_variable = { party_pop_array^v > election_threshold }
			}
			set_variable = { party_pop_elect_array^v = party_pop_array^v }
		}
		else_if = {
			set_variable = { party_pop_elect_array^v = 0 }
		}
	}
	update_majority_threshold = yes
	find_two_highest_parties = yes
}

add_up_party_totals = {

	set_variable = { pop_total_democratic = 0 }
	set_variable = { pop_total_communism = 0 }
	set_variable = { pop_total_fascism = 0 }
	set_variable = { pop_total_neutrality = 0 }
	set_variable = { pop_total_nationalist = 0 }
	
	#sum of ideologies
	for_loop_effect = {
		end = party_pop_array^num
		value = current_id
		meta_effect = {
			text = { add_to_variable = { pop_total_[PART_OF_IDEOLOGY1] = party_pop_array^current_id } }
			PART_OF_IDEOLOGY1 = "[?this.offcial_name_party_ideology]"
		}
	}
	#Maintain consistency with outlooks
	for_loop_effect = {
		end = party_pop_array^num
		value = current_id
		multiply_variable = { party_pop_array^current_id = 1000 }
		meta_effect = {
			text = {
				divide_variable = { party_pop_array^current_id = pop_total_[PART_OF_IDEOLOGY2] }
				multiply_variable = { party_pop_array^current_id = party_popularity@[PART_OF_IDEOLOGY2] }
			}
			PART_OF_IDEOLOGY2 = "[?this.offcial_name_party_ideology]"
		}
		round_variable = party_pop_array^current_id
		divide_variable = { party_pop_array^current_id = 1000 }
	}
}

set_party_positions = {
	resize_temp_array = { 
		array = party_pop_array_temp 
		size = 24
	}
	for_loop_effect = {
		end = pos_array^num
		find_highest_in_array = {
			array = party_pop_array 
			value = max
			index = max_index
		}
		set_variable = { pos_array^v = max_index }
		set_variable = { party_pop_array^max_index = -1 }
		set_temp_variable = { party_pop_array_temp^max_index = max }
	}
	clear_array = party_pop_array
	for_each_loop = {
		array = party_pop_array_temp
		add_to_array = { party_pop_array = v  }
	}
	clear_temp_array = party_pop_array_temp
}

find_two_highest_parties = {
	find_highest_in_array = {
		array = party_pop_elect_array
		value = max_1_tmp
		index = max_index_1_tmp
	}
	set_variable = { max_index_1 = max_index_1_tmp }
	set_variable = { party_pop_elect_array^max_index_1 = -1 }
	find_highest_in_array = {
		array = party_pop_elect_array
		value = max_2_tmp
		index = max_index_2_tmp
	}
	set_variable = { max_index_2 = max_index_2_tmp }
	set_variable = { party_pop_elect_array^max_index_1 = max_1_tmp }
}

set_coalition_drift = {
	#reset coalition counts
	set_variable = { coalition_fascism = 0 }
	set_variable = { coalition_nationalist = 0 }
	set_variable = { coalition_communism = 0 }
	set_variable = { coalition_neutrality = 0 }
	set_variable = { coalition_democratic = 0 }
	clear_coalition_count_ideas = yes
	
	for_loop_effect = {
		end = gov_coalition_array^num 
		value = v
		set_temp_variable = { sum = party_pop_elect_array^gov_coalition_array^v } #pop of current element
		set_temp_variable = { current_id = gov_coalition_array^v } #ID of current element
		
		meta_effect = {
			text = {
				add_to_variable = { coalition_[PART_OF_IDEOLOGY] = sum }
			}
			PART_OF_IDEOLOGY = "[offcial_name_party_ideology]"
		}
	}
	#prevents adding to own ideology
	meta_effect = {
		text = {
			set_variable = { coalition_[CURRENT_RULING_IDEOLOGY] = 0 }
		}
		CURRENT_RULING_IDEOLOGY = "[offcial_name_party_ideology]"
	}
	reset_democratic_coalition_ideas = yes
	reset_neutrality_coalition_ideas = yes
	reset_communism_coalition_ideas = yes
	reset_nationalist_coalition_ideas = yes
	reset_fascism_coalition_ideas = yes
}

update_government_coalition_strength = {
	hidden_effect = { gov_pop_idea_remove = yes }
	set_temp_variable = { sum = 0 }
	for_each_loop = {
		array = gov_coalition_array 
		value = v
		add_to_temp_variable = { sum = party_pop_array^v }
	}
	set_variable = { government_coalition_strength = sum }
	
	#strength of ruling_party is separate
	set_temp_variable = { sum = 0 }
	for_each_loop = {
		array = ruling_party
		value = v
		add_to_temp_variable = { sum = party_pop_array^v }
	}
	add_to_variable = { government_coalition_strength = sum }
	
	### Now setup the election strength ###
	set_temp_variable = { sum = 0 }
	for_each_loop = {
		array = gov_coalition_array 
		value = v
		add_to_temp_variable = { sum = party_pop_elect_array^v }
	}
	set_variable = { government_coalition_strength_elect = sum }
	
	#strength of ruling_party is separate
	set_temp_variable = { sum = 0 }
	for_each_loop = {
		array = ruling_party
		value = v
		add_to_temp_variable = { sum = party_pop_elect_array^v }
	}
	add_to_variable = { government_coalition_strength_elect = sum }
	
	#find the differentials
	set_variable = { government_coalition_differential = government_coalition_strength }
	subtract_from_variable = { government_coalition_differential = government_coalition_strength_elect }
	hidden_effect = { gov_pop_idea_add = yes }
	set_coalition_drift = yes
}

refund_government_coalition = {
	set_temp_variable = { sum_factor = 0 }
	for_loop_effect = {
		end = party_pop_array^num 
		value = v
		set_temp_variable = { sum = party_pop_elect_array^gov_coalition_array^v } #pop of current element
		set_temp_variable = { current_id = gov_coalition_array^v } #ID of current element
		meta_effect = {
			text = {
				pol_distance_matrix_[NUM] = yes
			}
			NUM = "[?current_id|.0]"
		}
		multiply_temp_variable = { sum = pol_distance_current }
		add_to_temp_variable = { sum_factor = sum }
	}
	set_variable = { pp_cost_to_invite = sum_factor }
	multiply_variable = { pp_cost_to_invite = 1500 } #75%
	add_political_power = pp_cost_to_invite
}

store_election_values = {

	for_loop_effect = {
		end = party_pop_array^num
		value = v
		if = {
			limit = {
				#true if no specific value has already been set
				check_variable = { party_pop_elect_array^v = -1 }
			}
			if = {
				limit = {
					#normal popularity entry is over threshold
					check_variable = { party_pop_array^v > election_threshold }
				}
				set_variable = { party_pop_elect_array^v = party_pop_array^v }
			}
			else_if = {
				limit = {
					check_variable = { party_pop_elect_array^v = -1 }
				}
				set_variable = { party_pop_elect_array^v = 0 }
			}
		}
	}
}

#only on startup
setup_election_threshold = {
	if = { limit = { NOT = { has_variable = election_threshold } }
		set_variable = { election_threshold = 0.05 } #5% used as default
	}
}

		set_temp_variable = { sum = 0 }

			add_to_temp_variable = { sum = v }
		}

}

reduce_affinity_array_party_1 = {
	
	for_loop_effect = {
		end = party_pop_array^num
		value = a1
		break = break_1
		set_temp_variable = { sum_coalition_1 = 0 }
		
		#current temp coalition1 strength
		for_each_loop = {
			array = party_pop_elect_array
			value = b1
			index = e1
			if = {
				limit = {
					check_variable = { party_1_affinity_array^e1 > 0 }
				}
				add_to_temp_variable = { sum_coalition_1 = b1 }
			}
		}
		if = {
			limit = {
				check_variable = { sum_coalition_1 > majority_threshold_1 }
			}
			find_highest_in_array = {
				array = party_1_affinity_array 
				value = max1
				index = max1_index
			}
			subtract_from_temp_variable = { sum_coalition_1 = party_pop_elect_array^max1_index }
			set_temp_variable = { party_1_affinity_array^max1_index = 0 }	
			if = {
				limit = {
					check_variable = { sum_coalition_1 < majority_threshold_1 }
				}
				set_temp_variable = { party_1_affinity_array^max1_index = max1 }
				add_to_temp_variable = { sum_coalition_1 = party_pop_elect_array^max1_index }
				set_temp_variable = { break_1 = 1 }
			}
		}
	}
	#find total cost1
	set_temp_variable = { cost1 = 0 }
	for_each_loop = {
		array = party_pop_elect_array
		value = d1
		index = g1
		if = {
			limit = {
				check_variable = { party_1_affinity_array^g1 > 0 }
			}
			set_temp_variable = { cost1_element = party_1_affinity_array^g1 }
			multiply_temp_variable = { cost1_element = d1 }
			add_to_temp_variable = { cost1 = cost1_element }
		}
	}
}
