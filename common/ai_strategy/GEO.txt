	allowed = { original_tag = GEO }
	enable = {
		NOT = { has_war_with = AFG }

		has_war_with = AFG

	ai_strategy = { type = support id = "AFG" value = 150 }
	ai_strategy = { type = send_volunteers_desire id = "AFG" value = 100 }
}
