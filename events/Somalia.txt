add_namespace = Somalia

### Somalia civil war cleanup events ###

#Give the winner SOM cosmetic tag
country_event = {
	id = Somalia.0
	
	hidden = yes
	
	trigger = {
		OR = {
			AND = {
				tag = SOM
				NOT = {
					country_exists = SHB 
				}
			}
			AND = {
				tag = SHB
				NOT = {
					country_exists = SOM 
				}
			}
		}
		NOT = { has_global_flag = Somalia_Civil_War_Over }
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = {
		#Civil war is over
		set_global_flag = Somalia_Civil_War_Over
		#Clear core status
		240 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		239 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		#Make the winner Somalia
		if = {
			limit = {
				tag = SHB
			}
			set_cosmetic_tag = SOM
		}
		add_state_core = 240
		add_state_core = 239
		set_capital = 240
	}
	
	option = {
		name = Somalia.0.a
	}
}

#Recreate Somalia in case it has been annexed
country_event = {
	id = Somalia.1
	
	hidden = yes
	
	trigger = {
		NOT = {
			country_exists = SOM
			country_exists = SHB
		}
		NOT = { 240 = { is_core_of = SOM } }
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = {
		#Clear core status
		240 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		239 = {
			remove_core_of = SOM
			remove_core_of = SHB
		}
		SOM = {
			add_state_core = 240
			add_state_core = 239
			set_capital = 240
		}
	}
	
	option = {
		name = Somalia.1.a
	}
}
