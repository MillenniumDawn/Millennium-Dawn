#Used only to intialize AB system
AB_init = {
	
	if = { 
		limit = {
			this = {
				has_defensive_war_with = event_target:AB_ATT
				has_country_flag = AB_deffender
			}
		}
		
		capital_scope = {
			add_claim_by = event_target:AB_ATT
		}
		
	}
}

#Used to check if AB System Active and
#attacker is still at war and assign event targets
AB_war_test = {

	if = { #checks if in war before proceeding.
		limit = {
			any_country = {
				has_defensive_war_with = PREV
				has_country_flag = AB_deffender
				capital_scope = {
					is_claimed_by = PREV
				}
			}
			PREV = {
				has_war = yes
				has_country_flag = AB_attacker
			}
		}
		PREV = {
			set_country_flag = AB_test_true
			save_event_target_as = AB_ATT
		}
		any_country = {
			limit = {
				has_defensive_war_with = PREV
				has_country_flag = AB_deffender
				capital_scope = {
					is_claimed_by = PREV
				}
			}
			save_event_target_as = AB_DEFF
		}
	}
	
	if = { #if not in war start returning national unity
		limit = {
			PREV = {
				not = { has_country_flag = test }
			}
		}
		#return national unity funcation
	}
	
}
