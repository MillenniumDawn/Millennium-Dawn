scripted_gui = {
	companies_slot = {
		window_name = "companies_slot"
        context_type = player_context
		parent_window_token = politics_tab

		effects = {
            companies_slot_btn_click = {
				if = {
					limit = {
						has_country_flag = CAC_companies_list
					}

					clr_country_flag = CAC_companies_list
					set_variable = { opened_company_details = -1 }
				}
				else = {
					set_country_flag = CAC_companies_list
				}
            }
		}
	}

    companies_list_window = {
		window_name = "companies_list_window"
        context_type = player_context
		parent_window_token = politics_tab

        visible = {
			ROOT = {
				AND = {
					is_ai = no
					has_country_flag = CAC_companies_list
				}
			}
		}
		
		triggers = {
			sort_size_click_enabled = {
				for_loop_effect = {
					start = 0
					end = CAC_companies^num
					value = i

					add_to_temp_array = {
						array = sorted_companies_list
						value = CAC_companies^i
					}

					add_to_temp_array = {
						array = sorted_size_companies_list
						value = CAC_companies_size^i
					}

					add_to_temp_array = {
						array = sorted_money_companies_list
						value = CAC_companies_money^i
					}

					add_to_temp_array = {
						array = sorted_value_companies_list
						value = CAC_companies_value^i
					}

					add_to_temp_array = {
						array = sorted_revenue_companies_list
						value = CAC_companies_revenue^i
					}
				}
			}
		}

		effects = {
			companies_details_button_click = {

				set_variable = { opened_company_details = v }
			}

			sort_size_click = {
				for_loop_effect = {
					start = 0
					end = CAC_companies^num
					value = i

					for_loop_effect = {
						start = 0
						end = CAC_companies^num
						value = j

						set_temp_variable = { next_j = j }
						add_to_temp_variable = { next_j = 1 }

						if = {
							limit = { check_variable = { sorted_size_companies_list^j < sorted_size_companies_list^next_j } }

							set_temp_variable = { temp = sorted_size_companies_list^j }
							set_temp_variable = { sorted_size_companies_list^j = sorted_size_companies_list^next_j }
							set_temp_variable = { sorted_size_companies_list^next_j = temp }

							set_temp_variable = { temp = sorted_companies_list^j }
							set_temp_variable = { sorted_companies_list^j = sorted_companies_list^next_j }
							set_temp_variable = { sorted_companies_list^next_j = temp }
						}
					}

					add_to_temp_array = {
						array = sorted_companies_list
						value = CAC_companies^size[i]
						index = i
					}
				}
			}

			sort_money_click = {
				for_loop_effect = {
					start = 0
					end = CAC_companies^num
					value = i

					for_loop_effect = {
						start = 0
						end = CAC_companies^num
						value = j

						set_temp_variable = { next_j = j }
						add_to_temp_variable = { next_j = 1 }

						if = {
							limit = { check_variable = { sorted_money_companies_list^j < sorted_money_companies_list^next_j } }

							set_temp_variable = { temp = sorted_money_companies_list^j }
							set_temp_variable = { sorted_money_companies_list^j = sorted_money_companies_list^next_j }
							set_temp_variable = { sorted_money_companies_list^next_j = temp }

							set_temp_variable = { temp = sorted_companies_list^j }
							set_temp_variable = { sorted_companies_list^j = sorted_companies_list^next_j }
							set_temp_variable = { sorted_companies_list^next_j = temp }
						}
					}

					add_to_temp_array = {
						array = sorted_companies_list
						value = CAC_companies^size[i]
						index = i
					}
				}
			}

			sort_value_click = {
			
			}

			sort_revenue_click = {

			}

			close_click = {
				clr_country_flag = CAC_companies_list
				set_variable = { opened_company_details = -1 }
            }
		}

		properties = {
			avaible_companies_grid_item_icon = {
				image = "[CAC_icon_of_company]"
			}
		}

        dynamic_lists = {
            avaible_companies_grid = {
                array = CAC_companies #sorted_companies_list
                entry_container = "avaible_companies_grid_list_item"
            }
        }
	}

				CAC_modify_factories_effect = yes
