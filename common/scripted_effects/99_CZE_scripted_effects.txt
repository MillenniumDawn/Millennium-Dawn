		var = CZE.skoda_support
		min = 0
		max = 100
	}
	clamp_variable = {
		var = CZE.volkswagen_support
		min = 0
		max = 100
	}

	if = {
		limit = {
			check_variable = { CZE.skoda_support = 100 }
		}
		CZE = {
			set_country_flag = CZE_skoda_overthrown_volkswagen_flag
		}

		limit = {

		}

		}
	}
}

		set_temp_variable = { base_price = 6 }

		set_temp_variable = { cze_revenue_low = 1.6875 }

		set_variable = { bought_generation = 2 }

		}
		else_if = {
			limit = { ROOT = { tag = ITA } }
			set_country_flag = { flag = CZE_skoda_cars_sold_to_italy days = 30 value = 1 }
		}
		else_if = {
			limit = { ROOT = { tag = SPR } }
			set_country_flag = { flag = CZE_skoda_cars_sold_to_spain days = 30 value = 1 }
		}
		else_if = {
			limit = { ROOT = { tag = POL } }
			set_country_flag = { flag = CZE_skoda_cars_sold_to_poland days = 30 value = 1 }
		}
		else_if = {
			limit = { ROOT = { tag = SOV } }
			set_country_flag = { flag = CZE_skoda_cars_sold_to_russia days = 30 value = 1 }
		}
		else_if = {

		}

			limit = { ROOT = { tag = CHI } }

		}
		else_if = {
			limit = { ROOT = { tag = TUR } }
			set_country_flag = { flag = CZE_skoda_cars_sold_to_turkey days = 30 value = 1 }
		}
		else_if = {
			limit = { ROOT = { tag = RAJ } }
			set_country_flag = { flag = CZE_skoda_cars_sold_to_india days = 30 value = 1 }
		}
		else_if = {
			limit = { ROOT = { tag = KOR } }
			set_country_flag = { flag = CZE_skoda_cars_sold_to_korea days = 30 value = 1 }
		}
		else_if = {
			limit = { ROOT = { tag = JAP } }
			set_country_flag = { flag = CZE_skoda_cars_sold_to_japan days = 30 value = 1 }

		else_if = {
			limit = { ROOT = { tag = HOL } }
			set_country_flag = { flag = CZE_skoda_cars_sold_to_netherlands days = 30 value = 1 }
		}

		# Add revenue based on demand
		CZE_skoda_get_country_demand_variable_effect = yes
		if = {
			limit = {

			add_to_variable = { CZE_skoda_superb_cars_treasury = cze_revenue_low }

			limit = {
