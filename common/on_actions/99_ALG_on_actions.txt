on_actions = {
    on_startup = {
        effect = {
            ALG = {
                country_event = { id = algeria_intro.1 }
			}
            KBY = {
                add_state_core = 385
            }
        }
    }
	on_weekly_ALG = {
		effect = {
			# If protests are ongoing and exhaustion is high, chance they end
			if = {
				limit = {
					has_country_flag = ALG_protests_ongoing
					check_variable = { ALG_protest_exhaustion > 19 }
				}
				random_list = {
					30 = { country_event = algerian_events.2025 } # End of protests
					70 = { } # Nothing happens
				}
			}

			# If protests are ongoing
			if = {
				limit = { has_country_flag = ALG_protests_ongoing }
				# Protests are not tired
			if = {
					limit = {
						check_variable = { ALG_protest_exhaustion < 10 }
					}
					random_list = {
						70 = { ALG_trigger_protest = yes }
						30 = { } # Nothing happens
					}
			}
					# Protests are starting to tire
				else_if = {
					limit = {
						check_variable = { ALG_protest_exhaustion > 9 }
					}
					random_list = {
						40 = { ALG_trigger_protest = yes }
						60 = { } # Nothing happens
					}
				}

				# Protests are tired
				else_if = {
					limit = {
						check_variable = { ALG_protest_exhaustion > 15 }
					}
					random_list = {
						20 = { ALG_trigger_protest = yes }
						80 = { } # Nothing happens
					}
				}
			}
		}
	}
	on_monthly_ALG = {
		effect = {
			random_list = {
				chance_to_kill_poachers = {
					subtract_from_variable = { no_of_poachers = no_of_poachers_to_kill }
					clamp_variable = { var = no_of_poachers min = 0 }
				}
				chance_to_not_kill_poachers = {
					add_to_variable = { no_of_poachers = 15 }
				}
			}
			if = {
				limit = {
					NOT = { has_country_flag = on_yearly_done_ALG }
					date > 2001.1.1
				}
				random_list = {
					sahara_expands = {
						country_event = { id = Algeria_pulse.1 }
					}
					sahara_steady = {
						country_event = { id = Algeria_pulse.2 }
					}
					sahara_retreats = {
						country_event = { id = Algeria_pulse.3 }
					}
				}
				set_country_flag = { flag = on_yearly_done_ALG days = 350 value = 1 }
			}
		}
	}

}
