Israel_Palestine_Hostile = {
	enable = {
		original_tag = ISR
		country_exists = HAM
		has_opinion_modifier = HAM_Are_Terrorists
	}
	abort = {
		OR = {
			NOT = { country_exists = HAM }
			NOT = { has_opinion_modifier = HAM_Are_Terrorists }
		}
	}

	ai_strategy = {
		type = contain
		id = "HAM"
		value = 50
	}
	ai_strategy = {
		type = antagonize
		id = "HAM"
		value = 100
	}
}
Israel_Palestine_Invade = {
	enable = {
		original_tag = ISR
		country_exists = HAM
		has_opinion_modifier = HAM_Are_Terrorists
		has_opinion = { target = HAM value < -90 }
	}
	abort = {
		OR = {
			NOT = { country_exists = HAM }
			NOT = { has_opinion_modifier = HAM_Are_Terrorists }
			has_opinion = { target = HAM value > -85 }
		}
	}

	ai_strategy = {
		type = conquer
		id = "HAM"
		value = 100
	}
}
