on_actions = {
    on_startup = {
        effect = {
            ALG = {
                country_event = { id = algeria_intro.1 }
			}
            KBY = {
                add_state_core = 385
            }
        }
    }
	on_weekly_ALG = {
		effect = {
			# If protests are ongoing and exhaustion is high, chance they end
			if = {
				limit = {
					has_country_flag = ALG_protests_ongoing
					check_variable = { ALG_protest_exhaustion > 19 }
				}
				random_list = {
					30 = { country_event = algerian_events.2025 } # End of protests
					70 = { } # Nothing happens
				}
			}

			# If protests are ongoing
			if = {
				limit = { has_country_flag = ALG_protests_ongoing }
				# Protests are not tired
			if = {
					limit = {
						check_variable = { ALG_protest_exhaustion < 10 }
					}
					random_list = {
						70 = { ALG_trigger_protest = yes }
						30 = { } # Nothing happens
					}
			}
					# Protests are starting to tire
				else_if = {
					limit = {
						check_variable = { ALG_protest_exhaustion > 9 }
					}
					random_list = {
						40 = { ALG_trigger_protest = yes }
						60 = { } # Nothing happens
					}
				}

				# Protests are tired
				else_if = {
					limit = {
						check_variable = { ALG_protest_exhaustion > 15 }
					}
					random_list = {
						20 = { ALG_trigger_protest = yes }
						80 = { } # Nothing happens
					}
				}
			}
		}
	}
	on_monthly_ALG = {
		if = {
			limit = {
				check_variable = { ALG_election_var > 0 }
			}
			add_to_variable = { var = ALG_election_var value = -1 }
			clamp_variable = { var = ALG_election_var min = 0 max = 48 }
		}

		if = {
			limit = {
				check_variable = { ALG_election_var = 0 }
			}
			add_to_variable = { var = ALG_election_var value = 48 }
			clamp_variable = { var = ALG_election_var min = 0 max = 48 }
			country_event = { id = ALG_pm_elections.1 days = 3 }
		}
		if = {
	 		limit = { check_variable = { ALG_pm_session_var > 0 } }
				add_to_variable = { var = ALG_pm_session_var value = -1 }
				clamp_variable = { var = ALG_pm_session_var min = 0 max = 48 }
		}
		if = {
			limit = { check_variable = { ALG_pm_session_var = 0 } }
				country_event = { id = ALG_pm.1 days = 1 }
		}


	}
}
