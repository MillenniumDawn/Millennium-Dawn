# These decisions are based around the Solomon Competition quest chain
# A chain of events focused on Australia that secretly builds up to a potential
# confrontation with China. If Australia neglects the Solomon Islands when offered
# these decisions, China will have the opportunity to step in. If this happens enough
# eventually China will have the opportunity to gain some sort of control over SOL.
SOL_developing_nation_category = {
	# Requesting industry investment - Approach Australia first
	SOL_AST_request_industry_investment = {
		icon = generic_construction
		priority = 20

		allowed = {
			original_tag = SOL
		}

		available = {
			AND = {
				NOT = { has_country_flag = SOL_CHI_requested_industry_construction }
				NOT = { has_country_flag = SOL_industry_construction_built }
			}
		}

		visible = {
			NOT = { has_country_flag = SOL_industry_construction_built }
		}

		fire_only_once = yes
		cost = 50

		complete_effect = {
			set_country_flag = SOL_AST_requested_industry_construction
			AST = {
				country_event = { id = solomon_competition.1 days = 5 }
			}
		}

		ai_will_do = {
			base = 5

			modifier = {
				has_country_flag = solomon_competition_AST_influence # Set in 99_AST_on_actions.txt
				factor = 2
			}
		}
	}

	# Requesting industry investment - Approach China first
	SOL_CHI_request_industry_investment = {
		icon = generic_construction
		priority = 20

		allowed = {
			original_tag = SOL
		}

		available = {
			AND = {
				NOT = { has_country_flag = SOL_AST_requested_industry_construction }
				NOT = { has_country_flag = SOL_industry_construction_built }
			}
		}

		visible = {
			NOT = { has_country_flag = SOL_industry_construction_built }
		}

		fire_only_once = yes
		cost = 50

		complete_effect = {
			set_country_flag = SOL_CHI_requested_industry_construction
			CHI = {
				country_event = { id = solomon_competition.1 days = 5 }
			}
		}

		ai_will_do = {
			base = 5

			modifier = {
				has_country_flag = solomon_competition_CHI_influence # Set in 99_AST_on_actions.txt
				factor = 2
			}
		}
	}
}
