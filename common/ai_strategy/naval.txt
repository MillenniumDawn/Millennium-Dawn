avoid_mediterranean = {
	#This is to avoid the Naval death trap that is Med without owning one of the islands or stats around the coast
	allowed = {
		NOT = {
			mediterranean_tags = yes
		}
	}
	enable = {
		has_war_with = { mediterranean_tags = yes }
		NOT = {
			controls_state = 19 #Gibraltar
			controls_state = 94 #Mallorca
			controls_state = 84 #Sardinia
			controls_state = 65 #Corscia
			controls_state = 84 #Malta\
			controls_state = 144 #Crete
			controls_state = 143 #Aegean Islands
			#Cyprus - 3 State
			controls_state = 145
			controls_state = 146
			controls_state = 812
		}

		alliance_naval_strength_ratio < 4

	abort_when_not_enabled = yes

	ai_strategy = { type = naval_avoid_region id = 25 value = 10000 }
