give_ai_mbt_template_0 = {
	if = {
		limit = {
			is_ai = yes
			NOT = { has_country_flag = country_has_mbt_0_templates }
		}

			name = "MBT 1965"
			type = mbt_hull_0

				main_armament_slot = tank_small_medium_cannon_2

				suspension_type_slot = tank_torsion_bar_suspension

				special_type_slot_1 = smoke_launchers

		}
	}
