add_namespace = greenland

## Alcohol Abuse Change
country_event = {
	id = greenland.1
	title = greenland.1.t
	desc = greenland.1.d
	picture = GFX_politics_protest
	is_triggered_only = yes

	trigger = {
		tag = GRN
		has_variable = {
			GRN_modify_alcohol_abuse
		}
	}

	option = {
		name = greenland.1.a
		custom_effect_tooltip = GRN_modify_alcohol_abuse_tt
	}
}
