# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes
#
defined_text = {
	name = GER_dark_winds_status
	text = {
		trigger = {
			check_variable = { GER_dark_winds_status > 0 }
			original_tag = GER
		}
		localization_key = GER_dark_winds_status_20
	}
	text = {
		trigger = {
			check_variable = { GER_dark_winds_status = 40 }
			original_tag = GER
		}
		localization_key = GER_dark_winds_status_40
	}
	text = {
		trigger = {
			check_variable = { GER_dark_winds_status = 60 }
			original_tag = GER
		}
		localization_key = GER_dark_winds_status_60
	}
    text = {
		trigger = {
			check_variable = { GER_dark_winds_status = 80 }
			original_tag = GER
		}
		localization_key = GER_dark_winds_status_80
	}
    text = {
		trigger = {
			check_variable = { GER_dark_winds_status > 80 }
			original_tag = GER
		}
		localization_key = GER_dark_winds_status_100
	}
}
