SOV_chechen_war = { #Prevents the CSTO from joining the Chechen War #helps Tensions
	allowed = { always = no } #No longer nesscary .CSTO is not a faction
	enable = {
		is_in_faction_with = SOV
		has_start_date < 2000.6.01
		country_exists = CHE
	}
	abort = {
		SOV = { owns_state = 672 }
		NOT = { country_exists = CHE }
	}

	ai_strategy = { type = dont_join_wars_with id = SOV target_country = CHE value = 500 }
	ai_strategy = { type = send_volunteers_desire id = SOV value = -500 }
}
