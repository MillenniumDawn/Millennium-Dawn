country_event = {
	id = worldwar.3
	immediate = { log = "[GetDateText]: [Root.GetName]: event worldwar.3" }
	title = worldwar.3.t
	desc = worldwar.3.d
	
	is_triggered_only = yes
	
	immediate = {
		USA = {
			if = {
				limit = { is_in_faction = yes }
				dismantle_faction = yes
			}
			create_faction = "American Alliance"
		}
		SOV = {
			if = {
				limit = { is_in_faction = yes }
				dismantle_faction = yes
			}
			create_faction = "Russian Alliance"
		}
		random_list = {
			50 = {
				USA = {
					declare_war_on = {
						target = SOV
						type = puppet_wargoal_focus
					}
				}
			}
			50 = {
				SOV = {
					declare_war_on = {
						target = USA
						type = puppet_wargoal_focus
					}
				}
			}
		}
		every_country = {
			if = {
				limit = {
					NOT = { tag = USA }
					NOT = { tag = SOV }
					NOT = { tag = CHI }
					has_war = no
				}
				if = {
					limit = {
						is_in_faction = yes
					}
					dismantle_faction = yes
				}
				random_list = {
					40 = {
						add_to_war = { 
							targeted_alliance = USA 
							enemy = SOV 
						}
					}
					40 = {
						add_to_war = { 
							targeted_alliance = SOV 
							enemy = USA 
						}
					}
					20 = { }
				}
			}
		}
		
	}
	
	option = {
		name = worldwar.3.a
		log = "[GetDateText]: [This.GetName]: worldwar.3.a executed"
	}
}
