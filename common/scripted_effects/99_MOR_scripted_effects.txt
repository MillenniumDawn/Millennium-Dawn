MOR_improved_credit_rating_effect = {
	if = {
		limit = { has_idea = MOR_extremely_poor_credit_rating_idea }
		swap_ideas = {
			remove_idea = MOR_extremely_poor_credit_rating_idea
			add_idea = MOR_slightly_better_credit_rating_idea
		}
	}
	else_if = {
		limit = { has_idea = MOR_slightly_better_credit_rating_idea }
		swap_ideas = {
			remove_idea = MOR_slightly_better_credit_rating_idea
			add_idea = MOR_improving_credit_rating_idea
		}
	}
	else_if = {
		limit = { has_idea = MOR_improving_credit_rating_idea }
		swap_ideas = {
			remove_idea = MOR_improving_credit_rating_idea
			add_idea = MOR_improved_credit_rating_idea
		}
	}
}
