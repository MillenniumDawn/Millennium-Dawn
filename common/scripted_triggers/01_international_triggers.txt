		has_country_flag = counterterror_open
	}
}

country_recognization_system_open = {

	}
}

pmc_system_open = {
	OR = {
		is_ai = yes
		has_country_flag = pmc_menu_open
	}
}

is_major_company = {
	OR = {
		has_country_flag = blackwater_sel
		has_country_flag = wagner_sel
		has_country_flag = aegis_sel
		has_country_flag = constellis_sel
		has_country_flag = md_sel
	}
}

is_full_state = {
	NOT = {
		has_idea = Non_State_Actor
		has_idea = Lacks_International_Recognition
		has_idea = rival_government
	}
}

pmc_can_recruit = {
	set_temp_variable = { max_merc_count = 1 }
	if = {
		limit = { check_variable = { gdp_total > 5000 } }
		set_temp_variable = { max_merc_count = 4 }
	}
	else_if = {
		limit = { check_variable = { gdp_total > 2000 } }
		set_temp_variable = { max_merc_count = 3 }
	}
	else_if = {
		limit = { check_variable = { gdp_total > 500 } }
		set_temp_variable = { max_merc_count = 2 }
	}

	custom_trigger_tooltip = {
		tooltip = pmc_surpassing_max_TT
		NOT = {
			# AI spams decisions
			check_variable = {
				var = ROOT.pmc_bought
				value = max_merc_count
				compare = greater_than_or_equals
			}
		}
	}
}
