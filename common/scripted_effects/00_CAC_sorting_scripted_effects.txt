CAC_sort_by_value_effect = {
    for_loop_effect = {
        start = 0
        end = CAC_companies_value^num
        value = i

        for_loop_effect = {
            start = 0
            end = companies_sorted_temp^num
            value = j

            if = {
                limit = { check_variable = { companies_sorted_temp^j < CAC_companies_value^i } }

                add_to_temp_array = {
                    array = companies_sorted_temp
                    value = CAC_companies_value^i
                    index = j
                }
            }
        }
    }
}
