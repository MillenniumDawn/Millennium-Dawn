# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = SWE_royal_acceptance_status
	text = {
		trigger = {
			check_variable = { SWE_royal_acceptance_status < 20 }
			original_tag = SWE
		}
		localization_key = SWE_royal_acceptance_status_low
	}
	text = {
		trigger = {
			check_variable = { SWE_royal_acceptance_status < 40 }
			original_tag = SWE
		}
		localization_key = SWE_royal_acceptance_status_moderat
	}
	text = {
		trigger = {
			check_variable = { SWE_royal_acceptance_status < 80 }
			original_tag = SWE
		}
		localization_key = SWE_royal_acceptance_status_high
	}
	text = {
		trigger = {
			check_variable = { var = SWE_royal_acceptance_status value = 80 compare = greater_than_or_equals }
			original_tag = SWE
		}
		localization_key = SWE_royal_acceptance_status_critical
	}
}
