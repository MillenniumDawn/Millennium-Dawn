BOS_update_ethnic_tensions = {
	if = { limit = { check_variable = { var = bosniak_happiness value = 100 compare = greater_than } } set_variable = { var = bosniak_happiness value = 100 } }
	if = { limit = { check_variable = { var = bosniak_happiness value = 0 compare = lesser_than } } set_variable = { var = bosniak_happiness value = 0 } }
	if = { limit = { check_variable = { var = croat_happiness value = 100 compare = greater_than } } set_variable = { var = croat_happiness value = 100 } }
	if = { limit = { check_variable = { var = croat_happiness value = 0 compare = lesser_than } } set_variable = { var = croat_happiness value = 0 } }
	if = { limit = { check_variable = { var = serbian_happiness value = 100 compare = greater_than } } set_variable = { var = serbian_happiness value = 100 } }
	if = { limit = { check_variable = { var = serbian_happiness value = 0 compare = lesser_than } } set_variable = { var = serbian_happiness value = 0 } }
	if = { limit = { check_variable = { var = rs_loyalty value = 100 compare = greater_than } } set_variable = { var = rs_loyalty value = 100 } }
	if = { limit = { check_variable = { var = rs_loyalty value = 0 compare = lesser_than } } set_variable = { var = rs_loyalty value = 0 } }
	if = { limit = { check_variable = { var = fbah_loyalty value = 100 compare = greater_than } } set_variable = { var = fbah_loyalty value = 100 } }
	if = { limit = { check_variable = { var = fbah_loyalty value = 0 compare = lesser_than } } set_variable = { var = fbah_loyalty value = 0 } }
}
