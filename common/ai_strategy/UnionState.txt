#UNION STATE IN ARMENIA CONTENT
#POLAND REVOULT (ARMENIA)
#UNION STATE SUBJECTS HELP REVOULT IN POLAND (MONARCHIST)
USR_subjects_help_arm_revoult_poland = {
	allowed = {
		OR = {
			original_tag = UKR
			original_tag = BLR
			original_tag = SER
		}
	}
	enable = {
		has_autonomy_state = autonomy_union_state 
		POL = {
			has_civil_war = yes
			is_subject_of = ARM
		}
		ARM = {	
			has_autonomy_state = autonomy_union_state 
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = support id = "POL" value = 200 }
	ai_strategy = { type = send_volunteers_desire id = "POL" value = 500 }
}
