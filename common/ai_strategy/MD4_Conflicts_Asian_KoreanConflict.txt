USA_SouthKorea_Relations = {
	enable = {
		original_tag = USA
		country_exists = KOR
		KOR = { has_government = democratic }
	}
	abort = {
		OR = {
			NOT = { country_exists = KOR }
			NOT = { country_exists = NKO }
			NOT = { KOR = { has_government = democratic } }
		}
	}
	ai_strategy = {
		type = befriend
		id = "KOR"
		value = 10
	}
	ai_strategy = {
		type = protect
		id = "KOR"
		value = 50
	}
}
USA_NorthKorea_Relations = {
	enable = {
		original_tag = USA
		country_exists = NKO
        has_opinion_modifier = NKO_Unrecognized_nuclear_power
	}
	abort = {
		OR = {
			NOT = { country_exists = NKO }
			NKO = { NOT = { has_opinion_modifier = NKO_Unrecognized_nuclear_power } }
			NOT = { NKO = { has_government = neutrality } }
		}
	}
	ai_strategy = {
		type = contain
		id = "NKO"
		value = 100
	}
	ai_strategy = {
		type = antagonize
		id = "NKO"
		value = 50
	}
}
