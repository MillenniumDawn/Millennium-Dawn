country_event = {
	id = russia.10
	immediate = { log = "[GetDateText]: [Root.GetName]: event russia.10" }
	title = {
		trigger = { BLR = { is_puppet_of = SOV } }
		text = russia.10.t1
	}
	title = {
		trigger = { BLR = { NOT = { is_puppet_of = SOV } } }
		text = russia.10.t2
	}
	desc = {
		trigger = { BLR = { is_puppet_of = SOV } } 
		text = russia.10.d1
	}
	desc = {
		trigger = { BLR = { NOT = { is_puppet_of = SOV } } }
		text = russia.10.d2
	}
	picture = GFX_report_event_russian_flag
	
	is_triggered_only = yes
	fire_only_once = yes 
	
	option = {
		name = russia.10.o1
		log = "[GetDateText]: [This.GetName]: russia.12.o1 executed"
		if = {
			limit = { BLR = { is_puppet_of = SOV } }
			BLR = { 
				every_owned_state = {
					add_core_of = SOV 
				}
			}
			annex_country = { target = SOV transfer_troops = yes }
		}
		if = {
			limit = { BLR = { NOT = { is_puppet_of = SOV } } }
			BLR = { 
				every_owned_state = {
					add_core_of = SOV 
				}
			}
			SOV = { puppet = BLR }
		}
		SOV = {
			country_event = { id = russia.3 days = 1 }
		}
		ai_chance = {
			base = 75
		}
	}
	
	option = { #Belarus Refuses!
		name = russia.10.o2
		log = "[GetDateText]: [This.GetName]: russia.10.o2 executed"
		if = {
			limit = { BLR = { is_puppet_of = SOV } }
			SOV = { release_puppet = BLR } 
			BLR = { 
				every_owned_state = {
					add_core_of = SOV 
				}
			}
			SOV = {
				create_wargoal = {
					target = BLR
					type = annex_everything 
				}
			}
		}
		if = {
			limit = { BLR = { NOT = { is_puppet_of = SOV } } }
			BLR = { 
				every_owned_state = {
					add_core_of = SOV 
				}
			}
			SOV = {
				create_wargoal = {
					target = BLR
					type = annex_everything 
				}
			}
		}
		SOV = { add_opinion_modifier = { target = BLR modifier = betrayed_our_cause } }
		SOV = { country_event = { id = russia.4 days = 1 } }
		ai_chance = {
			base = 25
		}
	}
