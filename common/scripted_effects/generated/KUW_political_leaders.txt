set_leader_KUW = {

	if = { limit = { has_country_flag = set_Western_Autocracy }

		if = { limit = { check_variable = { Western_Autocracy_leader = 0 } }
			set_variable = { Western_Autocracy_leader = 1 }
			kill_country_leader = yes

			create_country_leader = {
				name = "Jaber Al-Ahmad Al-Sabah"
				picture = "jaber_al_sabah.dds"
				ideology = Western_Autocracy
				traits = {
					ERROR
				}
			}

			if = { limit = { has_country_flag = do_not_retire } subtract_from_variable = {Western_Autocracy_leader = 1 } }
			if = { limit = { date < 2016.1.2 } set_temp_variable = { b = 1 } } #skip if 2017
		}
		if = { limit = { check_variable = { Western_Autocracy_leader = 1 NOT = { check_variable = { b = 1 } } } }
			set_variable = { Western_Autocracy_leader = 1 }
			kill_country_leader = yes

			create_country_leader = {
				name = "Sabah Al-Ahmad Al-Jaber Al-Sabah"
				picture = "Sabah_Al-Ahmad_Al-Jaber_Al-Sabah.dds"
				ideology = Western_Autocracy
				traits = {
					ERROR
				}
			}

			if = { limit = { has_country_flag = do_not_retire } subtract_from_variable = {Western_Autocracy_leader = 1 } }
			set_temp_variable = { b = 1 }
		}
	}
}
