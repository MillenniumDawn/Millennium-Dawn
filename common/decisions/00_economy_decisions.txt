# economy_decisions:
# Generic economic decisions for the player
economy_decisions = {
	lease_military_factories = {
		icon = ger_mefo_bills
		allowed = { always = yes }
		fire_only_once = no
		days_remove = 90
		days_re_enable = 50

		available = {
			NOT = { has_decision = lease_civilian_factories }
			NOT = { has_decision = lease_naval_factories }
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete_effect lease_military_factories"
			add_offsite_building = { level = 3 type = arms_factory }
			if = {
				limit = {
					has_idea = defense_industry
					NOT = { has_country_flag = recently_leased_military_factories }
				}
				set_country_flag = { flag = recently_leased_military_factories days = 185 value = 1 }

				# Modify Opinion
				if = { limit = { check_variable = { defense_industry_opinion > 0 } }
					set_temp_variable = { temp_opinion = -5 }
					change_defense_industry_opinion = yes
				}
			}
			custom_effect_tooltip = treasury_loss_lease_mils_TT
		}

		remove_effect = {
			add_offsite_building = { level = -3 type = arms_factory }
		}

		# Disabled AI for this Decision
	}

	lease_civilian_factories = {
		allowed = { always = yes }
		icon = ger_mefo_bills

		fire_only_once = no
		days_remove = 90
		days_re_enable = 30

		available = {
			NOT = { has_decision = lease_military_factories }

			log = "[GetDateText]: [Root.GetName]: Decision lease_civilian_factories"
			add_offsite_building = { level = 3 type = industrial_complex }
			if = {
				limit = {
					OR = {
						has_idea = industrial_conglomerates
						has_idea = the_donju
						has_idea = the_bazaar
						has_idea = chaebols
						has_idea = small_medium_business_owners
					}
					NOT = { has_country_flag = recently_supported_small_business }
				}
				set_country_flag = { flag = recently_supported_small_business days = 180 value = 1 }

				# Modify Opinion
				set_temp_variable = { temp_opinion = -5 }
				if = {
					limit = {
						has_idea = industrial_conglomerates
						check_variable = { industrial_conglomerates_opinion < 100 }
					}
					change_industrial_conglomerates_opinion = yes
				}
				if = {
					limit = {
						has_idea = small_medium_business_owners
						check_variable = { small_medium_business_owners_opinion < 100 }
					}
					change_small_medium_business_owners_opinion = yes
				}
				if = {
					limit = {
						has_idea = the_donju
						check_variable = { the_donju_opinion < 100 }
					}
					change_the_donju_opinion = yes
				}
				if = {
					limit = {
						has_idea = the_bazaar
						check_variable = { the_bazaar_opinion < 100 }
					}
					change_the_bazaar_opinion = yes
				}
				if = {
					limit = {
						has_idea = chaebols
						check_variable = { chaebols_opinion  < 100 }
					}
					change_chaebols_opinion = yes
				}
				custom_effect_tooltip = internal_faction_every_six_months_tt
			}
			custom_effect_tooltip = treasury_loss_lease_civs_TT
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove lease_civilian_factories"
			add_offsite_building = { level = -3 type = industrial_complex }
		}
		# Disabled AI for this Decision
	}

	temporary_tax_cuts = {
		icon = ger_mefo_bills
		allowed = { always = yes }
		available = {
			custom_trigger_tooltip = {
				check_variable = { population_tax_rate  > 3 }
				tooltip = per_tax_rate_at_least_3_tt
			}
			custom_trigger_tooltip = {
				check_variable = { var = population_tax_rate value = 50 compare = less_than_or_equals }
				tooltip = per_tax_rate_less_than_or_equals_50
			}
			hidden_trigger = {
				NOT = { has_country_flag = no_personal_tax_rate_change }
			}
		}

		fire_only_once = no
		days_remove = 120

		# Triggers Immediately
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision temporary_tax_cuts"
			hidden_effect = {
				subtract_from_variable = { population_tax_rate = 3 }
				ingame_update_setup = yes
				set_variable = { law_attitude = 0 } #index for taxes in #_law_change_attitude arrays
				set_variable = { law_change = -1 } #-1 decrease, 1 for increase
				law_attitude_change = yes
			}
			disable_personal_tax_rate_change = yes
			custom_effect_tooltip = reduce_per_tax_rate_3_tt
		}

		# After 120 days this fires
		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove temporary_tax_cuts"
			hidden_effect = {
				add_to_variable = { population_tax_rate = 3 }
				ingame_update_setup = yes
			}
			enable_personal_tax_rate_change = yes
			custom_effect_tooltip = add_per_tax_rate_3_tt
		}

		# Disabled AI for this decision
	}
