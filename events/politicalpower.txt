add_namespace = political_power
add_namespace = national_unity

#Check national unity, adjust political power
# Event has to be triggered for every country & it can't be hidden or the pp wont update for a month
country_event = {
	id = political_power.1
	title = political_power.1.t
	desc = "test"
	
	hidden = yes
	#fire_only_once = yes
	
	trigger = {
		#tag = GER #
		date > 1936.1.1
		not = {has_country_flag = pp_gain_set}
	}
	
	
	mean_time_to_happen = {
        days = 2
    }
	
	
	immediate  = {
		every_country = { 
			country_event = {id = political_power.2}
			country_event = {id = national_unity.1}
			
		}
	}
	option = {
		name = "test"
		#this = {gearz_test = yes}
	}
	
	
}
