on_actions = {
	on_monthly_IRQ = {
		effect = {
			if = {
				limit = {
					IRQ = {
						has_war_with = USA
					}
					IRQ = {
						has_country_flag = IRQ_iraq_war_ongoing
					}
				}
				add_to_variable = { var = IRQ_war_time_length value = -1 }
				clamp_variable = { var = IRQ_war_time_length min = 0 max = 5 }
			}
			if = {
				limit = {
					IRQ = {
						has_country_flag = IRQ_iraq_war_ongoing
					}
					check_variable = { from.IRQ_war_time_length < 1 }
				}
				USA = {
					country_event = {
						id = iraq_war.5
						days = 1
					}
				}
			}
			if = {
				limit = {
					USA = {
						has_country_flag = USA_IRQ_war_extended
					}
					check_variable = { from.IRQ_war_time_length < 1 }
				}
				USA = {
					country_event = {
						id = iraq_war.8
						days = 1
					}
				}
			}

			# Chance of Decrease in Readiness if not Conducting Attack

			if = {
				limit = {
					has_country_flag = IRQ_insurgency_active
					NOT = {
						has_country_flag = IRQ_conducting_attack
					}
				}
				random_list = {
					35 = {
						add_to_variable = { var = IRQ_government_readiness value = -1 }
						clamp_variable = { var = IRQ_government_readiness min = 0 max = 100 }
					}
					65 = {
						# Nothing Happens
					}
				}
			}

			# Attacks Being Conducted by Salafists against Player (Sets flag to Activate Mission)

			if = {
				limit = {
					NOT = {
						has_country_flag = IRQ_jihadist_conducting_attack
					}
				}
				35 = {
					# Nothing Happnes
				}
				65 = {
					IRQ = {
						set_country_flag = IRQ_jihadist_conducting_attack
					}
				}
			}

			# Attacks Being Conducted by the Shi'ite Forces if you aren't allied (Sets flag to activate mission)

			if = {
				limit = {
					NOT = {
						has_country_flag = IRQ_shias_conducting_attack_enemy
					}
				}
				35 = {
					# Nothing Happnes
				}
				65 = {
					IRQ = {
						set_country_flag = IRQ_shias_conducting_attack_enemy
					}
				}
			}

		}
	}
}
