### Multi Ethnic ###
country_event = {
	id = set_spirit.1
	title = set_spirit.1.t
	desc = set_spirit.1.d
	
	hidden = yes
	is_triggered_only = yes
	
	option = {
		name = set_spirit.1.a
		every_country = {	  
			limit = { Multi_Ethnic_State = yes }
			add_ideas = multi_ethnic_state_idea
		}
	}
}
