	target_trigger = {
		FROM = {
			exists = yes
		}
	}
	available = {
		ROOT = {
			custom_trigger_tooltip = {
				tooltip = spy_sat_mission_trigger_TT
				OR = {
					check_variable = { var_SPY_mission_num_available > FROM.num_controlled_states }
					check_variable = { var_SPY_mission_num_available = FROM.num_controlled_states }
				}
			}
		}
	}
	target_array = spy_mission_array
	fire_only_once = yes
	days_remove = 30
	targeted_modifier = {
		tag = FROM
		#attack_bonus_against = 0.1
		#defense_bonus_against = 0.1
		civilian_intel_decryption_bonus = 25
		army_intel_decryption_bonus = 25
		navy_intel_decryption_bonus = 25
		airforce_intel_decryption_bonus = 25
	}
	complete_effect = {
		ROOT = {
			add_to_variable = { var_SPY_mission_active_sat = FROM.num_controlled_states }
			update_SPY_mission_num = yes
		}
	}
	remove_effect = {
		ROOT = {
			subtract_from_variable = { var_SPY_mission_active_sat = FROM.num_controlled_states }
			update_SPY_mission_num = yes
		}
	}
}
