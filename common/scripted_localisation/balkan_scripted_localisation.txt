# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = Balkan_status
	text = {
		trigger = {
			check_variable = { balkan_stability < 21 }
			original_tag = SER
		}
		localization_key = balkan_stability_20
	}
	text = {
		trigger = {
			check_variable = { balkan_stability = 30 }
			original_tag = SER
		}
		localization_key = balkan_stability_30
	}
	text = {
		trigger = {
			check_variable = { balkan_stability = 40 }
			original_tag = SER
		}
		localization_key = balkan_stability_40
	}
    text = {
		trigger = {
			check_variable = { balkan_stability = 50 }
			original_tag = SER
		}
		localization_key = balkan_stability_50
	}
    text = {
		trigger = {
			check_variable = { balkan_stability > 60 }
			original_tag = SER
		}
		localization_key = balkan_stability_60
	}
}
