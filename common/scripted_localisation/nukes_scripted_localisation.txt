defined_text = {
	name = GerWarStatus
	
	text = {
		trigger = { owner = { has_war_with = ROOT } }
		localization_key = has_war_TT
	}
	text = {
		trigger = { owner = { NOT = { has_war_with = ROOT } } }
		localization_key = not_has_war_TT
	}
}

	name = GetBomberStatus

	}
