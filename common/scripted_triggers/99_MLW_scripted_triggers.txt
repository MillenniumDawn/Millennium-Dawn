MLW_MSCC_worker_cost_trigger = {
	if = {
		limit = {
			MLW = {
				has_completed_focus = MLW_gcc_megaprojects
				NOT = { has_completed_focus = MLW_gcc_constructors }
			}
		}
		set_temp_variable = { MLW_GCC_worker_cost_temp = 6 }
		if = {
			limit = {
				MLW = {
					OR = {
						has_idea = paralyzing_corruption
						has_idea = crippling_corruption
						has_idea = rampant_corruption
					}
				}
			}
			multiply_temp_variable = { MLW_GCC_worker_cost_temp = 0.4 }
		}
		else_if = {
			limit = {
				MLW = {
					OR = {
						has_idea = unrestrained_corruption
						has_idea = systematic_corruption
						has_idea = widespread_corruption
					}
				}
			}
			multiply_temp_variable = { MLW_GCC_worker_cost_temp = 0.8 }
		}
		else_if = {
			limit = {
				MLW = {
					OR = {
						has_idea = medium_corruption
						has_idea = modest_corruption
						has_idea = slight_corruption
					}
				}
			}
			multiply_temp_variable = { MLW_GCC_worker_cost_temp = 1.2 }
		}
		else = { multiply_temp_variable = { MLW_GCC_worker_cost_temp = 1.6 } }
	}
	else_if = {
		limit = {
			MLW = {
				has_completed_focus = MLW_gcc_megaprojects
				has_completed_focus = MLW_gcc_constructors
			}
		}
		set_temp_variable = { MLW_GCC_worker_cost_temp = 3 }
		if = {
			limit = {
				MLW = {
					OR = {
						has_idea = paralyzing_corruption
						has_idea = crippling_corruption
						has_idea = rampant_corruption
					}
				}
			}
			multiply_temp_variable = { MLW_GCC_worker_cost_temp = 0.8 }
		}
		else_if = {
			limit = {
				MLW = {
					OR = {
						has_idea = unrestrained_corruption
						has_idea = systematic_corruption
						has_idea = widespread_corruption
					}
				}
			}
			multiply_temp_variable = { MLW_GCC_worker_cost_temp = 1.6 }
		}
		else_if = {
			limit = {
				MLW = {
					OR = {
						has_idea = medium_corruption
						has_idea = modest_corruption
						has_idea = slight_corruption
					}
				}
			}
			multiply_temp_variable = { MLW_GCC_worker_cost_temp = 2.4 }
		}
		else = { multiply_temp_variable = { MLW_GCC_worker_cost_temp = 3.2 } }
	}
}
