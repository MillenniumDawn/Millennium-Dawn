  new_scripted_gui = {
    window_name = "cis_scripted_gui_window"
    context_type = player_context

    visible = {
		ROOT = {has_country_flag =  open_my_new_gui}
    }

    effects = {
    }

    triggers = {

    dynamic_lists = {
	}

    properties = {
    }

    ai_enabled = {
		always = no
    }

    ai_test_interval = 290  # The higher, the long it takes the ai to do actions
	ai_test_variance = 0.5

    ai_check = {
    }

    ai_check_scope = {
    }

    ai_weights = {
    }
  }

  cis_main_button = {
	context_type = player_context
	parent_window_token = politics_tab
	window_name = "cis_main_button"

	visible = {
		is_ai = no
	}

	ai_enabled = {
		always = no
	}

	effects = {
		cis_main_button_bg_click = {
			ROOT = {sound_effect = "menu_open_window"}
			if = {
				limit = { NOT = { has_country_flag = open_my_new_gui } }
				set_country_flag = open_my_new_gui
			}
			else = {
				clr_country_flag = open_my_new_gui
			}
		}
	}

}
