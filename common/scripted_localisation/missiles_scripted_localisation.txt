# Allows you to create your own dynamic keys
# to be called in localization.
# defined_text -> this is it, we're defining the text
# text -> a discrete entry that can be picked to display in loc.
# trigger -> determines if a text entry will be picked or not.
# (The triggers need to be valid for the scope the key is called in
# (eg Root or From.From).)
# localization_key -> points to the localization key
# that'll be used if trigger passes

defined_text = {
	name = missile_type_stat_loc
	text = {
		trigger = {
			check_variable = { missile_type_array^var_stat_trajectory_ID < 1 }
		}
		localization_key = missile_type_0
	}
	text = {
		trigger = {
			check_variable = { missile_type_array^var_stat_trajectory_ID = 1 }
		}
		localization_key = missile_type_1
	}
	text = {
		trigger = {
			check_variable = { missile_type_array^var_stat_trajectory_ID = 2 }
		}
		localization_key = missile_type_2
	}
	text = {
		trigger = {
			check_variable = { missile_type_array^var_stat_trajectory_ID = 3 }
		}
		localization_key = missile_type_3
	}
	text = {
		trigger = {
			check_variable = { missile_type_array^var_stat_trajectory_ID = 4 }
		}
		localization_key = missile_type_4
	}
	text = {
		trigger = {
			check_variable = { missile_type_array^var_stat_trajectory_ID = 5 }
		}
		localization_key = missile_type_5
	}
	text = {
		trigger = {
			check_variable = { missile_type_array^var_stat_trajectory_ID = 6 }
		}
		localization_key = missile_type_6
	}
}
