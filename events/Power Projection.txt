add_namespace = projection

#hidden event that adds flag
country_event = {
	id = projection.1
	
	hidden = yes
	
	trigger = {
		any_country = {
			has_military_access_to = ROOT
			NOT = { has_country_flag = military_base_@ROOT }
		}
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	option = {
		name = projection.1.a
		log = "[GetDateText]: [This.GetName]: projection.1.a executed"
		every_country = {
			limit = { has_military_access_to = ROOT }
			set_country_flag = military_base_@ROOT
		}
	}
}
