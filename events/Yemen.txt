# General Civil War Ending Event
news_event = {
    id = Yemen.4
    title = Yemen.4.t
    desc = Yemen.4.d
    picture = GFX_Yem_CivilWar_End
    
    fire_only_once = yes
	major = yes

    trigger = {
        # or means Any of the following must be true    
        or = {
            #If Houthis wins
            and = {
                tag = HOU
                NOT = {
                    country_exists = YEM
                    country_exists = AQY
                }
            }
           #If Yemen wins
            and = {
                tag = YEM
                NOT = {
                    country_exists = HOU
                    country_exists = AQY
                }
            }
           #If AQY Wins
            and = {
                tag = AQY
                NOT = {
                    country_exists = HOU
                    country_exists = YEM
                }
            }
			
        has_global_flag = Yemen_Civil_War_Over    
		
        }
    }
	
	mean_time_to_happen = {
		days = 0
	}
    
    #Option if Houthis wins
    option = {
        name = Yemen.4.a
        trigger = { original_tag  = HOU }
    }
    
    #Option if Yemen wins
    option = {
        name = Yemen.4.b
        trigger = { original_tag  = YEM }
    }
    #Option if Ansar al-Sharia wins
    option = {
        name = Yemen.4.c
        trigger = { original_tag  = AQY }
    }
	
	#Option for the rest of the world
    option = {
        name = Yemen.4.e
		trigger = {
			NOT = {
				TAG = YEM
				TAG = HOU
				TAG = AQY
			}
		}
    }
}
