# NCO_select_company
# Required Input: nco_selected_company (int)
# Purpose: Add the nco_selected_company to companys list
NCO_select_company_effect = {
    set_temp_variable = { selected_companies_num_temp = NCO_country_selected_companies^num }

    if = {
        limit = {
            check_variable = { selected_companies_num_temp < 3 }
        }

        add_to_array = {
		    array = NCO_country_selected_companies
			value = nco_selected_company
			index = selected_companies_num_temp
		}
		add_to_array = {
			array = NCO_country_selected_companies_type
			value = NCO_companies_type^nco_selected_company
			index = selected_companies_num_temp
		}

        add_to_array = {
			array = NCO_country_selected_companies_size
			value = 0
			index = selected_companies_num_temp
		}
        add_to_array = {
			array = NCO_country_selected_companies_development_points
			value = 1
			index = selected_companies_num_temp
		}
        add_to_array = {
			array = NCO_country_selected_companies_development_xp
			value = 100
			index = selected_companies_num_temp
		}

        # Economy
        add_to_array = {
			array = NCO_country_selected_companies_money
			value = 1
			index = selected_companies_num_temp
		}
        add_to_array = {
			array = NCO_country_selected_companies_value
			value = 6
			index = selected_companies_num_temp
		}

        add_to_array = {
			array = NCO_country_selected_companies_revenue
			value = 0.0
			index = selected_companies_num_temp
		}
        add_to_array = {
			array = NCO_country_selected_companies_income
			value = 0.0
			index = selected_companies_num_temp
		}
        add_to_array = {
			array = NCO_country_selected_companies_expenses
			value = 0.0
			index = selected_companies_num_temp
		}
        add_to_array = {
			array = NCO_country_selected_companies_factories
			value = 1
			index = selected_companies_num_temp
		}

        add_to_array = {
			array = NCO_country_selected_companies_income_from_factory
			value = 0.2
			index = selected_companies_num_temp
		}
        add_to_array = {
			array = NCO_country_selected_companies_expenses_from_factory
			value = 0.1
			index = selected_companies_num_temp
		}

        add_to_array = {
			array = NCO_country_selected_companies_additional_income
			value = 0
			index = selected_companies_num_temp
		}
        add_to_array = {
			array = NCO_country_selected_companies_additional_expenses
			value = 0
			index = selected_companies_num_temp
		}

        # company modifiers
        add_to_array = {
			array = NCO_company_modifier_political_power_factor
			value = -0.05
			index = selected_companies_num_temp
		}
        add_to_array = {
			array = NCO_company_modifier_consumer_goods_factor
			value = 0
			index = selected_companies_num_temp
		}

        set_temp_variable = { nco_political_power_to_remove_temp = GLOBAL.NCO_companies_cost^nco_selected_company }
        multiply_temp_variable = { nco_political_power_to_remove_temp = -1 }

        ROOT = { add_political_power = var:nco_political_power_to_remove_temp }

        log = "Selected company in [THIS.GetName]: slot-[?i] company_id-[?nco_selected_company]"
    }
}

# NCO_remove_company_by_slot
# Required Input: nco_slot_to_remove (int)
# Purpose: Remove company on slot nco_slot_to_remove
NCO_remove_company_by_slot_effect = {
    if = {
        limit = {
            AND = {
                check_variable = { nco_slot_to_remove > -1 }
            }
        }

        add_to_array = {
			array = NCO_country_avaibles_companies
			value = NCO_country_selected_companies^nco_slot_to_remove
		}

        remove_from_array = {
            array = NCO_country_selected_companies
            index = nco_slot_to_remove
        }
        remove_from_array = {
            array = NCO_country_selected_companies_type
            index = nco_slot_to_remove
        }
        remove_from_array = {
            array = NCO_country_selected_companies_size
            index = nco_slot_to_remove
        }
        remove_from_array = {
            array = NCO_country_selected_companies_development_points
            index = nco_slot_to_remove
        }
        remove_from_array = {
            array = NCO_country_selected_companies_development_xp
            index = nco_slot_to_remove
        }
        remove_from_array = {
            array = NCO_country_selected_companies_factories
            index = nco_slot_to_remove
        }
        remove_from_array = {
            array = NCO_country_selected_companies_additional_factories
            index = nco_slot_to_remove
        }

        if = {
            limit = {
                check_variable = { nco_slot_to_remove = 0 }
            }

            clear_array = NCO_country_selected_companies_upgrades_slot_0
        }
        else = {
            if = {
                limit = {
                    check_variable = { nco_slot_to_remove = 1 }
                }

                clear_array = NCO_country_selected_companies_upgrades_slot_1
            }
            else = {
                if = {
                    limit = {
                        check_variable = { nco_slot_to_remove = 2 }
                    }

                    clear_array = NCO_country_selected_companies_upgrades_slot_2
                }
            }
        }

        # company modifiers
        remove_from_array = {
            array = NCO_company_modifier_political_power_factor
            index = nco_slot_to_remove
        }
        remove_from_array = {
            array = NCO_company_modifier_consumer_goods_factor
            index = nco_slot_to_remove
        }
    }
}

NCO_update_companies_variables_effect = {
    if = {
        limit = {
            check_variable = { NCO_country_selected_companies^num > 0 }
        }

        for_loop_effect = {
            start = 0
            end = NCO_country_selected_companies^num
            value = i

            set_temp_variable = { company_income_temp = NCO_country_selected_companies_factories^i }
            multiply_temp_variable = { company_income_temp = NCO_country_selected_companies_income_from_factory^i }
            add_to_temp_variable = { company_income_temp = NCO_country_selected_companies_additional_income^i }

            set_variable = { NCO_country_selected_companies_income^i = company_income_temp }

            set_temp_variable = { company_expenses_temp = NCO_country_selected_companies_factories^i }
            multiply_temp_variable = { company_expenses_temp = NCO_country_selected_companies_expenses_from_factory^i }
            add_to_temp_variable = { company_expenses_temp = NCO_country_selected_companies_additional_expenses^i }

            set_variable = { NCO_country_selected_companies_expenses^i = company_expenses_temp }

            set_temp_variable = { company_new_money = company_money_to_change }
            
            set_variable = { NCO_country_selected_companies_revenue^i = company_money_to_change }
        }
    }
}

NCO_update_companies_upkeep_effect = {
    if = {
        limit = {
            check_variable = { NCO_country_selected_companies^num > 0 }
        }

        for_loop_effect = {
            start = 0
            end = NCO_country_selected_companies^num
            value = i

            set_temp_variable = { company_income_temp = NCO_country_selected_companies_factories^i }
            multiply_temp_variable = { company_income_temp = NCO_country_selected_companies_income_from_factory^i }
            add_to_temp_variable = { company_income_temp = NCO_country_selected_companies_additional_income^i }

            set_variable = { NCO_country_selected_companies_income^i = company_income_temp }

            set_temp_variable = { company_expenses_temp = NCO_country_selected_companies_factories^i }
            multiply_temp_variable = { company_expenses_temp = NCO_country_selected_companies_expenses_from_factory^i }
            add_to_temp_variable = { company_expenses_temp = NCO_country_selected_companies_additional_expenses^i }

            set_variable = { NCO_country_selected_companies_expenses^i = company_expenses_temp }

            set_temp_variable = { company_money_to_change = NCO_country_selected_companies_income^i }
            subtract_from_temp_variable = { company_money_to_change = NCO_country_selected_companies_expenses^i }

            set_temp_variable = { company_new_money = company_money_to_change }
            add_to_temp_variable = { company_new_money = NCO_country_selected_companies_money^i }

            set_variable = { NCO_country_selected_companies_revenue^i = company_money_to_change }

            if = {
                limit = {
                    check_variable = { company_new_money < 0 }
                }

                set_variable = { NCO_country_selected_companies_money^i = 0 }

                # there will be event what we doing now with company
                # add_to_variable = { NCO_country_selected_companies_value^i = company_money_to_change }
            }
            else = {
                add_to_variable = { NCO_country_selected_companies_money^i = company_money_to_change }
                add_to_variable = { NCO_country_selected_companies_value^i = company_money_to_change }
            }
        }
    }
}
