#Kurds rally for autonomy
country_event = {
    id = kurd_rally.1
    title = kurd_rally.1.t
    desc = kurd_rally.1.d
	picture = GFX_kurd_rally
	
	fire_only_once = yes
	
	trigger = {
	tag = TUR
	    TUR = {
	    controls_state = 162
	    has_idea = multi_ethnic_state_idea
            NOT = {
            has_country_flag = kurd_rally1
            }
        }
	    PKK = {
	    exists = no
	    }
	}

		mean_time_to_happen = {
		days = 1
	}

	#Ignore them
    option = {
    name = kurd_rally.1.a
	custom_effect_tooltip = kurd_rally_tt.1
	hidden_effect = {
	    TUR = {
		    set_country_flag = kurd_rally1
		    }
	    }
    }
}
