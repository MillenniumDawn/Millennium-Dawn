set_leader_SOM = {

	if = { limit = { has_country_flag = set_Neutral_conservatism }

		if = { limit = { check_variable = { Neutral_conservatism_leader = 0 } }
			set_variable = { Neutral_conservatism_leader = 1 }
			kill_country_leader = yes

			create_country_leader = {
				name = "Barre Adan Shire Hiiraale"
				picture = "SOM_Barre_Adan_Shire_Hiiraale.dds"
				ideology = Neutral_conservatism
				traits = {
				}
			}

			if = { limit = { has_country_flag = do_not_retire } subtract_from_variable = {Neutral_conservatism_leader = 1 } }
			if = { limit = { date < 2016.1.2 } set_temp_variable = { b = 1 } } #skip if 2017
		}
		if = { limit = { check_variable = { Neutral_conservatism_leader = 1 NOT = { check_variable = { b = 1 } } } }
			set_variable = { Neutral_conservatism_leader = 1 }
			kill_country_leader = yes

			create_country_leader = {
				name = "Abdiqasim Salad Hassan"
				picture = "SOM_Abdiqasim_Salad_Hassan.dds"
				ideology = Neutral_conservatism
				traits = {
				}
			}

			if = { limit = { has_country_flag = do_not_retire } subtract_from_variable = {Neutral_conservatism_leader = 1 } }
			if = { limit = { date < 2016.1.2 } set_temp_variable = { b = 1 } } #skip if 2017
		}
	}
}
