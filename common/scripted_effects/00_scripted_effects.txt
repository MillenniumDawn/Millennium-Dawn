set_NU_gain = {
	this = {set_country_flag = nu_gain_set}
	if = {
		limit = {
			this = {
				has_government = democratic
				and = {
					democratic > 0
					democratic < 0.1
				}
			} 
		}
		this = {
			add_national_unity = -0.01
			
		}
	}
	
	if = {
		limit = {
			this = {
				has_government = democratic
				and = {
					democratic > 0.09
					democratic < 0.2
				}
			} 
		}
		this = {
			add_national_unity = -0.007
		}
	}
	if = {
		limit = {
			this = {
				has_government = democratic
				and = {
					democratic > 0.19
					democratic < 0.3
				}
			} 
		}
		this = {
			add_national_unity = -0.005
		}
	}
	
	if = {
		limit = {
			this = {
				has_government = democratic
				and = {
					democratic > 0.29
					democratic < 0.4
				}
			} 
		}
		this = {
			add_national_unity = -0.002
		}
	}
	
	if = {
		limit = {
			this = {
				has_government = democratic

					democratic < 0.5
				}
			} 
		}

	}

			add_national_unity = 0.004
			
		}
	}
	
	if = {
		limit = {
			this = {
				has_government = democratic
				and = {
					democratic > 0.69
					democratic < 0.8
				}
			} 
		}
		this = {
			add_national_unity = 0.006
		}
	}
	
	if = {
		limit = {
			this = {

					democratic > 0.79
					democratic < 0.9
				}
			} 
		}

			add_national_unity = 0.008

		}
	}
