# The 2005 West African Drought
country_event = {
	id = african.1
	title = african.1.t
	desc = african.1.d
	picture = GFX_nomadic_arabic_people
	is_triggered_only = yes
	trigger = {
		OR = {
			original_tag = MAL
			original_tag = NGR
			original_tag = BFA
		}
	}

	# Unfortunate
	option = {
		name = african.1.a
		log = "[GetDateText]: [This.GetName]: african.1.a executed"
		add_stability = -0.02
		set_temp_variable = { temp_opinion = -5 }
		change_farmers_opinion = yes

		# Mali Custom Stuff
		if = { limit = { original_tag = MAL }
			country_event = { id = mali.2 days = 7 random_days = 14 }
		}

		ai_chance = {
			base = 1
		}
	}

	# Send in Support
	option = {
		name = african.1.b
		log = "[GetDateText]: [This.GetName]: african.1.b executed"
		set_temp_variable = { treasury_change = gdp_total }
		multiply_temp_variable = { treasury_change = -0.003 }
		modify_treasury_effect = yes
		set_temp_variable = { temp_opinion = 5 }
		change_farmers_opinion = yes

		# Mali Custom Stuff
		if = { limit = { original_tag = MAL }
			country_event = { id = mali.2 days = 7 random_days = 14 }
		}

		ai_chance = {
			base = 1
		}
	}
}

		}

		multiply_temp_variable = { treasury_change = -0.003 }
		modify_treasury_effect = yes

		change_farmers_opinion = yes

		set_party_index_to_ruling_party = yes
