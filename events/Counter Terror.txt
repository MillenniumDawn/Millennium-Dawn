	hidden = yes

	mean_time_to_happen = { days = 2 }

	trigger = {
		OR = {
			AND = {
				NOT = { has_idea = foreign_military_advisors1 }
				check_variable = { foreign_advisors > 0 }
				check_variable = { foreign_advisors < 6 }
			}
			AND = {
				NOT = { has_idea = foreign_military_advisors2 }
				check_variable = { foreign_advisors > 5 }
				check_variable = { foreign_advisors < 11 }
			}
			AND = {
				NOT = { has_idea = foreign_military_advisors3 }
				check_variable = { foreign_advisors > 10 }
			}
			AND = {
				OR = {
					has_idea = foreign_military_advisors1
					has_idea = foreign_military_advisors2
					has_idea = foreign_military_advisors3
				}
				check_variable = { foreign_advisors = 0 }
			}
		}
	}

	option = {
		name = advisors.1.a
		log = "[GetDateText]: [This.GetName]: advisors.1.a executed"
		if = {
			limit = {
				NOT = { has_idea = foreign_military_advisors1 }
				check_variable = { foreign_advisors > 0 }
				check_variable = { foreign_advisors < 6 }
			}
			if = {
				limit = { has_idea = foreign_military_advisors2 }
				swap_ideas = {
					remove_idea = foreign_military_advisors2
					add_idea = foreign_military_advisors1
				}
			}
			else_if = {
				add_ideas = foreign_military_advisors1
			}
		}
		else_if = {
			limit = {
				NOT = { has_idea = foreign_military_advisors2 }
				check_variable = { foreign_advisors > 5 }
				check_variable = { foreign_advisors < 11 }
			}
			if = {
				limit = {
					NOT = {
						has_idea = foreign_military_advisors1
						has_idea = foreign_military_advisors3
					}
				}
				add_ideas = foreign_military_advisors2
			}
			else_if = {
				limit = { has_idea = foreign_military_advisors1 }
				swap_ideas = {
					remove_idea = foreign_military_advisors1
					add_idea = foreign_military_advisors2
				}
			}
			else_if = {
				limit = { has_idea = foreign_military_advisors3 }
				swap_ideas = {
					remove_idea = foreign_military_advisors3
					add_idea = foreign_military_advisors2
				}
			}
		}
		else_if = {
			limit = {
				NOT = { has_idea = foreign_military_advisors3 }
				check_variable = { foreign_advisors > 10 }
			}
			if = {
				limit = {
					NOT = {
						has_idea = foreign_military_advisors1
						has_idea = foreign_military_advisors2
					}
				}
				add_ideas = foreign_military_advisors3
			}
			if = {
				limit = { has_idea = foreign_military_advisors1 }
				swap_ideas = {
					remove_idea = foreign_military_advisors1
					add_idea = foreign_military_advisors3
				}
			}
			else_if = {
				limit = { has_idea = foreign_military_advisors2 }
				swap_ideas = {
					remove_idea = foreign_military_advisors2
					add_idea = foreign_military_advisors3
				}
			}
		}
		else_if = {
			limit = { check_variable = { foreign_advisors = 0 } }
			if = {
				limit = { has_idea = foreign_military_advisors1 }
				remove_ideas = foreign_military_advisors1
			}
			else_if = {
				limit = { has_idea = foreign_military_advisors2 }
				remove_ideas = foreign_military_advisors1
			}
			else_if = {
				limit = { has_idea = foreign_military_advisors3 }
				remove_ideas = foreign_military_advisors1
			}
		}
	}
}
