    title = HOL_Benelux_Air.33.t
    desc = HOL_Benelux_Air.33.d
    picture = GFX_report_event_generic_air

    is_triggered_only = yes

    option = {
        name = HOL_Benelux_Air.33.a
        log = "[GetDateText]: [This.GetName]: HOL_Benelux_Air.33.a executed"
        hidden_effect = {
            if = {
                limit = {
                    has_global_flag = NLD_accepts_mrtt
                    has_global_flag = BEL_accepts_mrtt
                    has_global_flag = LUX_accepts_mrtt
                }
				set_temp_variable = { treasury_change = 7.5 }
				modify_treasury_effect = yes
            }
            else_if = {
                limit = {
                    OR = {
                        AND = { has_global_flag = NLD_accepts_mrtt has_global_flag = BEL_accepts_mrtt }
                        AND = { has_global_flag = NLD_accepts_mrtt has_global_flag = LUX_accepts_mrtt }
                        AND = { has_global_flag = BEL_accepts_mrtt has_global_flag = LUX_accepts_mrtt }
                    }
                }
				set_temp_variable = { treasury_change = 4.5 }
				modify_treasury_effect = yes
            }
        }
    }
}
