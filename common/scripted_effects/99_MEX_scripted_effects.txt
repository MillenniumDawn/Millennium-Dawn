MEX_increase_tank_research = {
	if = {
		limit = {
			has_idea = MEX_tank_negligence_2
		}
		remove_ideas = MEX_tank_negligence_2
	}
	else_if = {
		limit = {
			has_idea = MEX_tank_negligence_1
		}
		swap_ideas = {
			remove_idea = MEX_tank_negligence_1
			add_idea = MEX_tank_negligence_2
		}
	}
	else_if = {
		limit = {
			has_idea = MEX_tank_negligence
		}
		swap_ideas = {
			remove_idea = MEX_tank_negligence
			add_idea = MEX_tank_negligence_1
		}
	}
}
