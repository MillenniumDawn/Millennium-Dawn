War_On_Terrorism_ISIS = {
	enable = {
		NOT = {
			tag = ISI
		}
        has_opinion_modifier = ISIS_Are_Terrorists
	}
	abort = {
        OR = {
            has_completed_focus = isis_allegiance
            NOT = { has_opinion_modifier = ISIS_Are_Terrorists }
        }
	}

	# ISIS
	ai_strategy = {
		type = antagonize
		id = "ISI"
		value = 100
	}
	ai_strategy = {
		type = contain
		id = "ISI"
		value = 100
	}
}
