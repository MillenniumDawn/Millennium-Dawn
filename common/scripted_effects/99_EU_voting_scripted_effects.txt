EU_update_AI_general_voting_modifiers = {	
	hidden_effect = {
		log = "[GetDateText]: [FROM.GetTag]: 99_EU_voting_scripted_effects EU_update_AI_general_voting_modifiers"	
		every_country = {
			limit = {
				has_idea = EU_member
			}
			set_variable = {	
				var = EU_AI_general_voting_modifier_YES
				value = 0
			}
			### modifiers for voting cheats
			if = {
				limit = {
					has_global_flag = EU_voting_cheat_yes
				}			
				add_to_variable = {
					var = EU_AI_general_voting_modifier_YES
					value = 1000 ### ai chance 					
				}
			}
			if = {
				limit = {
					has_global_flag = EU_voting_cheat_no
				}			
				add_to_variable = {
					var = EU_AI_general_voting_modifier_YES
					value = -1000 ### ai chance
				}
			}

			### modifiers for influence of ROOT on all other EU members 
			if = {
				limit = {
					influence_on_all_other_EU_members_5_percent = yes
				}			
				add_to_variable = {
					var = EU_AI_general_voting_modifier_YES
					value = 5 ### ai chance yes +5 no -5
				}
			}
			if = {
				limit = {
					influence_on_all_other_EU_members_10_percent = yes
				}			
				add_to_variable = {
					var = EU_AI_general_voting_modifier_YES
					value = 5 ### ai chance yes +5 no -5
				}
			}
			if = {
				limit = {
					influence_on_all_other_EU_members_20_percent = yes
				}			
				add_to_variable = {
					var = EU_AI_general_voting_modifier_YES
					value = 10 ### ai chance yes +10 no -10
				}

					var = EU_AI_general_voting_modifier_YES

				}

					value = 15 ### ai chance yes +15 no -15
				}

			### modifiers for eurosceptic
			if = {
				limit = {
					THIS = {	
						check_variable = {
							var = THIS.eurosceptic
							value = 0.1
							compare = greater_than_or_equals

					}
				}			
				add_to_variable = {
					var = EU_AI_general_voting_modifier_YES
					value = -5 ### ai chance yes -5 no +5
				}
			}
			if = {
				limit = {
					THIS = {	
						check_variable = {
							var = THIS.eurosceptic
							value = 0.2
							compare = greater_than_or_equals
						}
					}
				}			
				add_to_variable = {
					var = EU_AI_general_voting_modifier_YES
					value = -5 ### ai chance yes -5 no +5
				}
			}
			if = {
				limit = {
					THIS = {	
						check_variable = {
							var = THIS.eurosceptic

					var = EU_AI_general_voting_modifier_YES
					value = -20 ### ai chance yes -20 no +20
				}
