sleeper_cell_decisions = {
	build_sleeper_cells = {
	
		icon =
	
		allowed = {
			always = yes
		}
	
		visible = {
			tag = ISI
			NOT = {
				FROM = {
					has_idea = Isis_sleeper_cell_idea
				}
			}
		}
		
		target_trigger = {
			exists = yes
			FROM = {
				AND = {
					NOT = { tag = ISI }
					NOT = { has_idea = Isis_sleeper_cell_idea }
					NOT = { is_puppet_of = ISI }
					NOT = { is_in_faction_with = ISI }
				}
			}
		}
		
		remove_effect = {
			FROM = {
				add_ideas = Isis_sleeper_cell_idea
			}
		}
		
		days_remove = 1
		fire_only_once = no
	}
}
