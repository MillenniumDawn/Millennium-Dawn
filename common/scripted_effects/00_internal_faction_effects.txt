# Economic Factions
display_election_status_small_medium_business_owners = {
	if = {
		limit = {
			has_idea = small_medium_business_owners
			hostile_small_medium_business_owners = yes
		}
		custom_effect_tooltip = display_election_hostile_smbo_status_tt
		add_to_variable = { campaign_funding_count = -2 }
	}
	else_if = {
		limit = {
			has_idea = small_medium_business_owners
			negative_small_medium_business_owners = yes
		}
		custom_effect_tooltip = display_election_negative_smbo_status_tt
		add_to_variable = { campaign_funding_count = -1 }
	}
	else_if = {
		limit = {
			has_idea = small_medium_business_owners
			indifferent_small_medium_business_owners = yes
		}
		custom_effect_tooltip = display_election_indifferent_smbo_status_tt
	}
	else_if = {
		limit = {
			has_idea = small_medium_business_owners
			positive_small_medium_business_owners = yes
		}
		custom_effect_tooltip = display_election_positive_smbo_status_tt
		add_to_variable = { campaign_funding_count = 1 }
	}
	else_if = {
		limit = {
			has_idea = small_medium_business_owners
			enthusiastic_small_medium_business_owners = yes
		}
		custom_effect_tooltip = display_election_enthusiastic_smbo_status_tt
		add_to_variable = { campaign_funding_count = 2 }
	}
}
