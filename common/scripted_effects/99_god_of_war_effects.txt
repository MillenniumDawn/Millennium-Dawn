ai_add_xp = {
	set_temp_variable = { buf = 0 }
	random_other_country = {
		limit = { is_ai = no }
		set_temp_variable = { buf = THIS }
	}

	set_temp_variable = { army_points = var:buf.army_doctrines_researched }
	set_temp_variable = { navy_points = var:buf.navy_doctrines_researched }
	set_temp_variable = { air_points = var:buf.air_doctrines_researched }
	set_temp_variable = { special_points = var:buf.special_doctrines_researched }
	add_to_temp_variable = { army_points = 5 }
	add_to_temp_variable = { navy_points = 3 }
	add_to_temp_variable = { navy_points = 3 }
	add_to_temp_variable = { special_points = 5 }

	if = {
		limit = {
			check_variable = { THIS.army_doctrines_researched < army_points }
		}
		army_experience = 50
	}
	if = {
		limit = {
			check_variable = { THIS.navy_doctrines_researched < navy_points }
		}
		navy_experience = 50
	}
	if = {
		limit = {
			check_variable = { THIS.air_doctrines_researched < air_points }
		}
		air_experience = 50
	}
	if = {
		limit = {
			check_variable = { THIS.special_doctrines_researched < special_points }
		}
		army_experience = 50
	}
}
