# Boost Alabama
country_event = {
	id = electoral_college.1
	immediate = { log = "[GetDateText]: [Root.GetName]: event electoral_college.1" }
	title = electoral_college.1.t
	desc = electoral_college.1.d

	is_triggered_only = yes

	option = {
		name = electoral_college.1.a
		log = "[GetDateText]: [This.GetName]: electoral_college.1.a executed"
		random_list = {
			75 = {
				if = {
					limit = { check_variable = { democrat_support^1 < 100 } }
					add_to_variable = { democrat_support^1 = 1 }
					add_to_variable = { republican_support^1 = -1 }
				}
			}
			25 = {
				if = {
					limit = { check_variable = { democrat_support^1 > 0 } }
					add_to_variable = { democrat_support^1 = -1 }
					add_to_variable = { republican_support^1 = 1 }
				}
			}
		}
		custom_effect_tooltip = left_leaning_boost_TT
	}
	option = {
		name = electoral_college.1.b
		log = "[GetDateText]: [This.GetName]: electoral_college.1.b executed"
		random_list = {
			75 = {
				if = {
					limit = { check_variable = { democrat_support^1 > 0 } }
					add_to_variable = { democrat_support^1 = -1 }
					add_to_variable = { republican_support^1 = 1 }
				}

# Election concludes. Determine winner
country_event = {
	id = electoral_college.53
	immediate = { 
		log = "[GetDateText]: [Root.GetName]: event electoral_college.53"
		# So players can't just delay the results
		determine_swing_states = yes 
		random_electoral_college_drift = yes
		us_party_faction_drift = yes
		update_us_subideologies = yes
		update_electral_college_alignments = yes
		if = {
			limit = { check_variable = { democrat_electoral_votes > republic_electoral_votes } }
			set_variable = { admin_elect = 0 }
			# Determines winning faction
			if = {
				limit = { 
					check_variable = { progressive_democrat > establishment_democrat }
					check_variable = { progressive_democrat > socialist_democrat } 
				}
				set_variable = { faction_elect = 2 }
			}
			else_if = {
				limit = { 
					check_variable = { socialist_democrat > establishment_democrat }
					check_variable = { socialist_democrat > progressive_democrat } 
				}
				set_variable = { faction_elect = 3 }
			}
			else = { set_variable = { faction_elect = 1 } }

			if = {
				# President won re-election, see y'all in 4 years
				limit = { 
					check_variable = { admin_elect = current_admin }
					check_variable = { terms = 1 }
				}
				add_to_variable = { terms = 1 }
				country_event = { id = electoral_college.54 days = 1 }
			}
			else_if = {
				# President lost re-election pepehands
				limit = { 
					NOT = { check_variable = { admin_elect = current_admin } }
					check_variable = { terms = 1 }
				}
				set_variable = { current_admin = 0 }
				country_event = { id = electoral_college.55 days = 1 }
			}
			else = { country_event = { id = electoral_college.56 days = 1 } }

		}
		else_if = {
			limit = { check_variable = { democrat_electoral_votes < republic_electoral_votes } }
			set_variable = { admin_elect = 1 }
			# Determines winning faction
			if = {
				limit = { 
					check_variable = { constitutionalist_republican > establishment_democrat }
					check_variable = { constitutionalist_republican > socialist_democrat } 
				}
				set_variable = { faction_elect = 2 }
			}
			else_if = {
				limit = { 
					check_variable = { socialist_democrat > establishment_democrat }
					check_variable = { socialist_democrat > constitutionalist_republican } 
				}
				set_variable = { faction_elect = 3 }
			}
			else = { set_variable = { faction_elect = 1 } }

			if = {
				# President won re-election, see y'all in 4 years
				limit = { 
					check_variable = { admin_elect = current_admin }
					check_variable = { terms = 1 }
				}
				add_to_variable = { terms = 1 }
				country_event = { id = electoral_college.54 days = 1 }
			}
			else_if = {
				# President lost re-election pepehands
				limit = { 
					NOT = { check_variable = { admin_elect = current_admin } }
					check_variable = { terms = 1 }
				}
				set_variable = { current_admin = 1 }
				country_event = { id = electoral_college.55 days = 1 }
			}
			else = { country_event = { id = electoral_college.57 days = 1 } }
		}
		# Lets just hope there isn't a tie cause I don't want to deal with that shit lmao. TODO ADD TIE CASE
	}

	is_triggered_only = yes
	hidden = yes

	trigger = {
		tag = USA
	}

	option = {
		name = electoral_college.53.a
		log = "[GetDateText]: [This.GetName]: electoral_college.53.a executed"
	}
}
