# How to use this system?
# set_leader_AST effect is called common/99_election_effects.txt
# set_leader_AST checks which political party is in power,
# then calls the elect_party effect for the relevant party.
# The elect_party effect contains the logic for choosing the current leader.
# Each leader has their own effect called to switch them in.
# retire_leader_AST effect is called in common/decisions/Political Desicions.txt
# retire_leader_AST adds the current leader to the retired_political_leaders array,
# which is currently setup in history/countries/AST - Australia.txt, although ideally
# integrating into the old system's effects would be better.
# Once a politician's ID is in the retired array, leadership will auto pass to the next
# eligible leader in the party, repeating until a non-retired leader is reached.

## Liberal Party - Index: 1 ##
elect_party_western_conservatism_AST = {
	if = {
		limit = {
			OR = { date > 2000.1.1 date = 2000.1.1 }
			date < 2007.1.1
		}
		elect_leader_john_howard_AST = yes
	}
	else_if = {
		limit = {
			OR = { date > 2007.1.1 date = 2007.1.1 }
			date < 2010.1.1
		}
		elect_leader_peter_costello_AST = yes
	}
	else_if = {
		limit = {
			OR = { date > 2010.1.1 date = 2010.1.1 }
			date < 2013.1.1
		}
		elect_leader_malcolm_turnbull_AST = yes
	}
	else_if = {
		limit = {
			OR = { date > 2013.1.1 date = 2013.1.1 }
			date < 2016.1.1
		}
		elect_leader_tony_abbott_AST = yes
	}
	else_if = {
		limit = { # Malcolm Turnbull got a second try after the Abbott disaster
			OR = { date > 2016.1.1 date = 2016.1.1 }
			date < 2018.1.1
		}
		elect_leader_malcolm_turnbull_AST = yes
	}
	else_if = {
		limit = {
			OR = { date > 2018.1.1 date = 2018.1.1 }
			date < 2022.1.1
		}
		elect_leader_scott_morrison_AST = yes
	}
	else_if = {
		limit = {
			date > 2022.1.1 date = 2022.1.1
		}
		elect_leader_peter_dutton_AST = yes
	}
}

# John Howard - Liberal Party - 0 #
elect_leader_john_howard_AST = {
	if = {
		limit = { is_in_array = { retired_political_leaders = 0 } }
		elect_leader_peter_costello_AST = yes
	}
	else = {
		set_variable = { current_political_leader = 0 }
		create_country_leader = {
			name = "John Howard"
			picture = "John_Howard.dds"
			desc = "JOHN_HOWARD_LEADER_DESC"
			ideology = conservatism
			traits = { western_conservatism capable lawyer pro_american }
		}
	}
}

### Set Leader - Decide political leader based on ruling party ###
set_leader_AST = {
	hidden_effect = { kill_country_leader = yes }
	## Western Outlook Parties ##

	# Liberal Party - Index: 1 #
	if = {
		limit = { has_country_flag = set_conservatism }
		elect_party_western_conservatism_AST = yes
	}

	# Centre Alliance - Index: 2 #
	else_if = {
		limit = { has_country_flag = set_liberalism }
		elect_party_western_liberalism_AST = yes
	}

	# Labor Party - Index: 3 #
	else_if = {
		limit = { has_country_flag = set_socialism }
		elect_party_western_socialism_AST = yes
	}

	## End of Western Outlook Parties ##

	## Emerging Outlook Parties ##

	# Communist Party of Australia (CPA) - Index: 4 #
	else_if = {
		limit = { has_country_flag = set_Communist-State }
		elect_party_emerging_communist_AST = yes
	}

	# Socialist Alliance - Index: 5 #
	else_if = {
		limit = { has_country_flag = set_anarchist_communism }
		elect_party_emerging_anarchist_communism_AST = yes
	}

	# Katter's Australian Party (KAP) - Index: 6 #
	else_if = {
		limit = { has_country_flag = set_Conservative }
		elect_party_emerging_conservative_AST = yes
	}

	## End of Emerging Outlook Parties ##

	## Non-aligned Outlook Parties ##

	# National Party of Australia - Index: 14 #
	else_if = {
		limit = { has_country_flag = set_Neutral_conservatism }
		elect_party_neutrality_neutral_conservatism_AST = yes
	}

	# Liberal Democratic Party (LDP) - Index: 16 #
	else_if = {
		limit = { has_country_flag = set_Neutral_Libertarian }
		elect_party_neutrality_neutral_libertarian_AST = yes
	}

	# Australian Greens - Index: 17 #
	else_if = {
		limit = { has_country_flag = set_Neutral_green }
		elect_party_neutrality_neutral_green_AST = yes
	}

	# Socialist Equality Party (SEP) - Index: 19 #
	else_if = {
		limit = { has_country_flag = set_Neutral_Communism }
		elect_party_neutrality_neutral_communism_AST = yes
	}

	## End of Non-aligned Outlook Parties ##

	## Nationalist Outlook Parties ##

	# One Nation - Index: 20 #
	else_if = {
		limit = { has_country_flag = set_Nat_Populism }
		elect_party_nationalist_nat_populism_AST = yes
	}

	# Australia First Party - Index: 21 #
	else_if = {
		limit = { has_country_flag = set_Nat_Fascism }
		elect_party_nationalist_nat_fascism_AST = yes
	}

	# Australian Defence Force - Index: 22 #
	else_if = {
		limit = { has_country_flag = set_Nat_Autocracy }
		elect_party_nationalist_nat_autocracy_AST = yes
	}

	## End of Nationalist Outlook Parties ##
	else = {
		log = "[GetDateText]: [Root.GetName]: Invalid ruling party for set_leader."
	}
}
