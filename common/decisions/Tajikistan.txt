TAJ_radicalism_system = {
	TAJ_deploy_national_guard = {
		allowed = { original_tag = TAJ }

		custom_cost_trigger = {
			command_power > 34
		}
		custom_cost_text = command_power_more_than_35

		available = {
			OR = {
				has_idea = defence_03
				has_idea = defence_04
				has_idea = defence_05
				has_idea = defence_06
				has_idea = defence_07
				has_idea = defence_08
				has_idea = defence_09
			}
			hidden_trigger = {
				command_power > 34
			}
		}

		visible = {
			NOT = {
				has_idea = TAJ_national_guard_deployed
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision TAJ_deploy_national_guard"
			hidden_effect = {
				add_command_power = -35
			}
			add_ideas = TAJ_national_guard_deployed
			newline = yes
			custom_effect_tooltip = TAJ_nat_guard_TT
		}

		ai_will_do = {
			factor = 10
		}
	}

	TAJ_recall_national_guard = {
		allowed = { original_tag = TAJ }

		visible = {
			has_idea = TAJ_national_guard_deployed
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision TAJ_recall_national_guard"
			remove_ideas = TAJ_national_guard_deployed
		}

		ai_will_do = {
			factor = 10
		}
	}

	TAJ_ban_the_irp = {
		allowed = { original_tag = TAJ }

		visible = {
			NOT = {
				has_idea = TAJ_flight_of_labor
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision TAJ_ban_the_irp"
			custom_effect_tooltip = TAJ_irp_ended_TT
			set_country_flag = TAJ_radicalism_ended
			add_to_variable = { var = TAJ_radical_influence value = -100 }
			clamp_variable = { var = TAJ_radical_influence min = 0 max = 100 }
		}

		ai_will_do = {
			factor = 10
		}
	}
}
TAJ_radicalism_system_2 = {
	TAJ_open_menu = {
		allowed = { original_tag = TAJ }

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision TAJ_open_menu"
			if = { limit = { NOT = { has_country_flag = international_open  } }
				set_country_flag = international_open
			}
			else = {
				clr_country_flag = international_open
			}
			if = {
				limit = {
					NOT = {
						has_country_flag = pmc_menu_open
						has_country_flag = state_menu_open
						has_country_flag = counterterror_open
					}
				}
				set_country_flag = pmc_menu_open
			}
			update_international_system_gui = yes
		}

		ai_will_do = {
			factor = 10
		}
	}

	TAJ_arrest_iskp = {

			}
			custom_trigger_tooltip = {
				tooltip = TAJ_ter_TT
				has_country_flag = threat_level_negligible
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision TAJ_arrest_iskp"
			remove_ideas = TAJ_isik
		}

		ai_will_do = {
			factor = 10
		}
	}
