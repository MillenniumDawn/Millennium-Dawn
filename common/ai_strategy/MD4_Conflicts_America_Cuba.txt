USA_Cuba_Relations = {
	enable = {
		original_tag = USA
		OR = {
			CUB = { has_government = communism }
			CUB = { has_government = fascism }
			CUB = { has_government = neutrality }
			AND = {
				has_opinion_modifier = embargo
				has_opinion_modifier = CUB_USA_Embargo
			}
		}
	}
	abort = {
		OR = {
			NOT = { country_exists = CUB }
			OR = {
				CUB = { NOT = { has_government = communism } }
				CUB = { NOT = { has_government = fascism } }
				CUB = { NOT = { has_government = neutrality } }
			}
			AND = {
				NOT = { has_opinion_modifier = embargo }
				NOT = { has_opinion_modifier = CUB_USA_Embargo }
			}
		}
	}
	ai_strategy = {
		type = contain
		id = "CUB"
		value = 100
	}
	ai_strategy = {
		type = antagonize
		id = "CUB"
		value = 50
	}
}

	enable = {
		tag = CUB
		
		OR = {
			has_opinion_modifier = embargo

		}
	}
	abort = {
		NOT = { country_exists = USA }
		USA = { NOT = { has_opinion_modifier = embargo } }
	}
	ai_strategy = {
		type = antagonize
		id = "USA"
		value = 100
	}
}
