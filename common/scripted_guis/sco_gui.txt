scripted_gui = {
	SCO_member_list_windows = {
		context_type = national_focus_context
		window_name = "SCO_member_list_windows"
		parent_window_token = national_focus
		visible = {
			original_tag = CHI
		}

		dirty = SCO_page_number
		# 0 means empty, 1 - Asia, 2 - Middle East, 3 - Africa, 4 - Americas 

		effects = {
			SCO_member_list_buttom_asia_click = {
				if = {
					limit = {
						check_variable = {
							var = SCO_page_number
							value = 1
							compare = not_equals
						}
					}
					set_variable = { SCO_page_number = 1 }
				}
				else = {
					set_variable = { SCO_page_number = 0 }
				}
			}

			SCO_member_list_buttom_middle_east_click = {
				if = {
					limit = {
						check_variable = {
							var = SCO_page_number
							value = 2
							compare = not_equals
						}
					}
					set_variable = { SCO_page_number = 2 }
				}
				else = {
					set_variable = { SCO_page_number = 0 }
				}
			}

			SCO_member_list_buttom_africa_click = {
				if = {
					limit = {
						check_variable = {
							var = SCO_page_number
							value = 3
							compare = not_equals
						}
					}
					set_variable = { SCO_page_number = 3 }
				}
				else = {
					set_variable = { SCO_page_number = 0 }
				}
			}

			SCO_member_list_buttom_south_america_click = {
				if = {
					limit = {
						check_variable = {
							var = SCO_page_number
							value = 4
							compare = not_equals
						}
					}
					set_variable = { SCO_page_number = 4 }
				}
				else = {
					set_variable = { SCO_page_number = 0 }
				}
			}
		}


		ai_enabled = { always = no }
	}
}
