# An Equal of An Enemy
country_event = {
	id = southern_republic.7
	title = southern_republic.7.t
	desc = southern_republic.7.d
	fire_only_once = yes
	is_triggered_only = yes

	# Yes
	option = {
		name = southern_republic.7.a
		log = "[GetDateText]: [This.GetName]: southern_republic.7.a executed"
		CSA = { add_to_faction = BRA }
		every_country = {
			if = {
				limit = {
					has_idea = rio_pact_member
				}
				remove_ideas = rio_pact_member
			}
		}
	}

	# No
	option = {
		name = southern_republic.7.b
		log = "[GetDateText]: [This.GetName]: southern_republic.7.b executed"
		CSA = {
			add_political_power = -100
			add_war_support = 0.05
			country_event = {
				id = southern_republic.3
				days = 1
			}
		}
	}
}
