CUB_update_military_standing = {
	set_variable = { CUB_military_standing_value = 0 }
	if = {
		limit = { has_idea = depression }
		add_to_variable = { CUB_update_military_standing = -20 }
	}
	else_if = {
		limit = { has_idea = stagnation }
		add_to_variable = { CUB_update_military_standing = -40 }
	}
	else_if = {
		limit = { has_idea = recession }
		add_to_variable = { CUB_update_military_standing = -60 }
	}
	else_if = {
		limit = { has_idea = stable_growth}
		add_to_variable = { CUB_update_military_standing = -80 }
	}
	else_if = {
		limit = { has_idea = fast_growth }
		add_to_variable = { CUB_update_military_standing = -100 }
	}
}
