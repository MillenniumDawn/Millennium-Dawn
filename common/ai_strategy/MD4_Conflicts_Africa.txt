Eritrea_Djibouti_Border_Conflict ={
	enable = {
		tag = ERI
        country_exists = DJI
        has_government = Nationalist
        has_opinion_modifier = ERI_Border_Disputes
	}
	abort = {
        OR = {
            NOT = { country_exists = DJI }
            NOT = { has_government = Nationalist }
            NOT = { has_opinion_modifier = ERI_Border_Disputes }
        }

	}
	
	ai_strategy = {
		type = antagonize
		id = "DJI"
		value = 50
	}

}
