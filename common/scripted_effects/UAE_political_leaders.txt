set_leader_UAE = {

	if = { limit = { has_country_flag = set_Kingdom }

		if = { limit = { check_variable = { Kingdom_leader = 0 } }
			set_variable = { Kingdom_leader = 1 }
			kill_country_leader = yes

			create_country_leader = {
				name = "Zayed bin Sultan Al-Nahyan"
				picture = "Zayed_bin_Sultan_al-Nahyan.dds"
				ideology = Kingdom
				traits = {
					king
					salafist_Kingdom
				}
			}

			if = { limit = { has_country_flag = do_not_retire } subtract_from_variable = {Kingdom_leader = 1 } }
			if = { limit = { date < 2016.1.2 } set_temp_variable = { b = 1 } } #skip if 2017
		}
		if = { limit = { check_variable = { Kingdom_leader = 1 NOT = { check_variable = { b = 1 } } } }
			set_variable = { Kingdom_leader = 1 }
			kill_country_leader = yes

			create_country_leader = {
				name = "Khalifa bin Zayed Al-Nahyan"
				picture = "Khalifa_bin_Zayed_Al-Nahyan.dds"
				ideology = Kingdom
				traits = {
					king
					salafist_Kingdom
					tech_savy
					greedy
					capable
				}
			}

			if = { limit = { has_country_flag = do_not_retire } subtract_from_variable = {Kingdom_leader = 1 } }
			set_temp_variable = { b = 1 }
		}
	}
}
