			has_country_flag = PER_election_enabled

			date > 2005.2.1

			date > 2013.2.1
			date > 2017.2.1
			date > 2021.2.1
		}
	}
	is_triggered_only = yes

	timeout_days = 1
	immediate = {

			country_event = {
				id = PER_election_events.2
				days = 0
			}
		}
		else_if = { #9th Period

				date > 2005.1.1
				date < 2006.1.1
			}
			country_event = {
				id = PER_election_events.3
				days = 0
			}
		}
		else_if = { #10th Period
			limit = {
				date > 2009.1.1
				date < 2010.1.1
			}
			country_event = {
				id = PER_election_events.4
				days = 0
			}
		}
		else_if = { #11th Period
			limit = {
				date > 2013.1.1
				date < 2014.1.1
			}
			country_event = {
				id = PER_election_events.5
				days = 0
			}
		}
		else_if = { #12th Period
			limit = {
				date > 2017.1.1
				date < 2018.1.1
			}

				date < 2022.1.1
