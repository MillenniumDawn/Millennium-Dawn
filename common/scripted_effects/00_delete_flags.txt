		else = { if = { limit = { has_idea = recession }
			set_country_flag = economic_cycle_5
		
			else = { if = { limit = { has_idea = stagnation }
				set_country_flag = economic_cycle_4
			
				else = { if = { limit = { has_idea = stable_growth }
					set_country_flag = economic_cycle_3
			
					else = { if = { limit = { has_idea = fast_growth }

				set_country_flag = generic_edu_3_flag

				} }
