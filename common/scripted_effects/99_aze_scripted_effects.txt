modify_aliev_suppport = {
	add_to_variable = { AZE.aliev_cult = arg1 }

	clamp_variable = {
		var = AZE.aliev_cult
		min = -100
		max = 100
	}

	if = {
		limit = {
			AZE = { has_idea = AZE_aliev_cult_very_weak }
		}

		modify_aliev_very_weak_support = yes
	}
	else_if = {
		limit = {
			AZE = { has_idea = AZE_aliev_cult_weak }
		}

		modify_aliev_weak_support = yes
	}
	else_if = {
		limit = {
			AZE = { has_idea = AZE_aliev_cult_medium }
		}

		modify_aliev_medium_support = yes
	}
	else_if = {
		limit = {
			AZE = { has_idea = AZE_aliev_cult_strong }
		}

		modify_aliev_strong_support = yes
	}
	else_if = {
		limit = {
			AZE = { has_idea = AZE_aliev_cult_very_strong }
		}

		modify_aliev_very_strong_support = yes
	}
}

modify_aliev_very_weak_support = {
	if = {
		limit = { check_variable = { AZE.aliev_cult < -60 } }
		# return
	}
	else_if = {
		limit = { check_variable = { AZE.aliev_cult < -20 } }

		swap_ideas = {
			remove_idea = AZE_aliev_cult_very_weak
			add_idea = AZE_aliev_cult_weak
		}

		AZE = { country_event = azeev.18 }
	}
	## we should not ever change by more then 10?
}

aze_increase_democratic_support = {
	if = {
		limit = { has_idea = AZE_westerndrift_idea5 }
		add_political_power = 200
	}
	else_if = {
		limit = { has_idea = AZE_westerndrift_idea4 }
		swap_ideas = {
			remove_idea = AZE_westerndrift_idea4
			add_idea = AZE_westerndrift_idea5
		}
	}
	else_if = {
		limit = { has_idea = AZE_westerndrift_idea3 }
		swap_ideas = {
			remove_idea = AZE_westerndrift_idea3
			add_idea = AZE_westerndrift_idea4
		}
	}
	else_if = {
		limit = { has_idea = AZE_westerndrift_idea2 }
		swap_ideas = {
			remove_idea = AZE_westerndrift_idea2
			add_idea = AZE_westerndrift_idea3
		}
	}
	else_if = {
		limit = { has_idea = AZE_westerndrift_idea }
		swap_ideas = {
			remove_idea = AZE_westerndrift_idea
			add_idea = AZE_westerndrift_idea2
		}
	}
	else = {
		add_ideas = AZE_westerndrift_idea
	}
}
