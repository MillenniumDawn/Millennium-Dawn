scripted_gui = {

	ep_window_button = {
		window_name = "ep_window_button_container"
		
		context_type = player_context
		parent_window_token = politics_tab
		
		visible = {
			always = yes
		}
		
		ai_enabled = {
			always = no
		}
		
		triggers = {	
			open_menu_button_click_enabled = {
				always = yes
			}
		}
		
		effects = {
			open_menu_button_click = {
				ROOT = {
					if = {
						limit = { NOT = { has_country_flag = ep_menu_opened } }
						set_country_flag = ep_menu_opened
					}
					else = {
						clr_country_flag = ep_menu_opened
					}
				}
			}
		}
	}
	
	ep_window_menu = {
		window_name = "ep_window_menu_container"
		
		context_type = player_context
		#parent_window_token = politics_tab
		
		visible = {
			has_country_flag = ep_menu_opened
		}
	}
}
