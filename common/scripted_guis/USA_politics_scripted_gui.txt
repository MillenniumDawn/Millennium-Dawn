scripted_gui = {
	USA_politics_open_button = {
		context_type = player_context
		parent_window_token = politics_tab
		window_name = "USA_politics_open_button"

		visible = { original_tag = USA }
		ai_enabled = { always = no }

		effects = {
			USA_politics_open_button_click = {
				if = {
					limit = { NOT = { has_country_flag = USA_politics_menu_opened } }
					set_country_flag = USA_politics_menu_opened
				}
				else = { clr_country_flag = USA_politics_menu_opened }
			}
		}
	}

	USA_politics_main_menu = {
		context_type = player_context
		window_name = "USA_politics_main_menu"

		properties = {}

		visible = { has_country_flag = USA_politics_menu_opened }

		effects = {
			USA_politics_close_button_click = {
				clr_country_flag = USA_viewing_congressional_makeup
				clr_country_flag = USA_viewing_senatorial_makeup
				clr_country_flag = USA_viewing_electoral_college_results
				clr_country_flag = USA_viewing_policies
				clr_country_flag = USA_politics_menu_opened
			}
			USA_view_congress_button_click = {
				set_country_flag = USA_viewing_congressional_makeup
				clr_country_flag = USA_viewing_senatorial_makeup
				clr_country_flag = USA_viewing_electoral_college_results
				clr_country_flag = USA_viewing_policies
			}
			USA_view_senate_button_click = {
				set_country_flag = USA_viewing_senatorial_makeup
				clr_country_flag = USA_viewing_congressional_makeup
				clr_country_flag = USA_viewing_electoral_college_results
				clr_country_flag = USA_viewing_policies
			}
			USA_view_electoral_college_button_click = {
				set_country_flag = USA_viewing_electoral_college_results
				clr_country_flag = USA_viewing_congressional_makeup
				clr_country_flag = USA_viewing_senatorial_makeup
				clr_country_flag = USA_viewing_policies
			}
			USA_view_policies_button_click = {
				set_country_flag = USA_viewing_policies
				clr_country_flag = USA_viewing_congressional_makeup
				clr_country_flag = USA_viewing_senatorial_makeup
				clr_country_flag = USA_viewing_electoral_college_results
			}
		}
	}
