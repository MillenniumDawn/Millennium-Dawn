	}

	}

			name = "Sergei Bagapsh"
			picture = "Sergei_Bagapsh.dds"
			ideology = Neutral_conservatism

				neutrality_Neutral_conservatism

			if = { limit = { NOT = { is_in_array = { ruling_party = 14 } } }

			}

			set_temp_variable = { temp_outlook_increase = 0.10 }
