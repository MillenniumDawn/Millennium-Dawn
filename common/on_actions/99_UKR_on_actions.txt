on_actions = {
	on_weekly_UKR = {
		effect = {
		###########KHARKIV TANK PROM TO UKRAINE###########	
		#KHARKIV TANK PROM  ADDED TO UKR
		if = {
			limit = {
				original_tag = UKR
				NOT = { has_idea = HPR_concern_sov }
				country_exists = HPR
				HPR = {
					is_subject_of = UKR
					has_idea = HPR_concern
				}
			}
			add_ideas = HPR_concern_sov
		}
		#KHARKIV TANK PROM  REMOVED FROM UKR
		if = {
			limit = {
				original_tag = UKR
				has_idea = HPR_concern_sov
				country_exists = HPR
				HPR = {
					NOT = { is_subject_of = UKR }
					has_idea = HPR_concern
				}
			}
			remove_ideas = HPR_concern_sov
		}
		#KHARKIV TANK PROM  WITHOUT HPR ADDED TO UKR
		if = {
			limit = {
				original_tag = UKR
				NOT = { has_idea = HPR_concern_sov }
				owns_state = 696
				NOT = { country_exists = HPR }
				HPR = { has_completed_focus = HPR_concers }
			}
			add_ideas = HPR_concern_sov
		}
		#KHARKIV TANK PROM  REMOVED FROM UKR WITHOUT HPR
		if = {
			limit = {
				original_tag = UKR
				has_idea = HPR_concern_sov
				NOT = { owns_state = 696 }
				NOT = { country_exists = HPR }
				CHE = { has_completed_focus = HPR_concers }
			}
			remove_ideas = HPR_concern_sov
		}
		###########CRIMEA MARINE PLANT OR UKR###########	
			#CRIMEA MARINE PLANT ADDED TO UKR
			if = {
				limit = {
					original_tag = UKR
					NOT = { has_idea = CRM_naval }
					country_exists = CRM
					CRM = {
						is_subject_of = UKR
						has_idea = CRM_naval
					}
				}
				add_ideas = CRM_naval
			}
			#CRIMEA MARINE PLANT REMOVED FROM UKR
			if = {
				limit = {
					original_tag = UKR
					has_idea = CRM_naval
					country_exists = CRM
					CRM = {
						NOT = { is_subject_of = UKR }
						has_idea = CRM_naval
					}
				}
				remove_ideas = CRM_naval
			}
			#CRIMEA MARINE PLANT ADDED TO UKR WITHOUT CRM
			if = {
				limit = {
					original_tag = UKR
					NOT = { has_idea = CRM_naval }
					owns_state = 1112
					NOT = { country_exists = CRM }
					CRM = { has_completed_focus = CRM_zavod_ships }
				}
				add_ideas = CRM_naval
			}
			#CRIMEA MARINE PLANT REMOVES FROM UKR WITHOUT CRM
			if = {
				limit = {
					original_tag = UKR
					has_idea = CRM_naval
					NOT = { owns_state = 1112 }
					NOT = { country_exists = CRM }
					CRM = { has_completed_focus = CRM_zavod_ships }
				}
				remove_ideas = CRM_naval
			}
    	}
    }
}
