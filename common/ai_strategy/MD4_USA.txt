## Politics

USA_KOR = { # South Korea
	enable = {
		tag = USA
		KOR = { has_government = democratic }
		country_exists = KOR
	}
	abort = {
		NOT = { country_exists = KOR }
		OR = {
			NOT = { KOR = { has_government = democratic } }
            NOT = { country_exists = NKO }
		}
	}
	ai_strategy = {
		type = protect
		id = "KOR"
		value = 200
	}
}
